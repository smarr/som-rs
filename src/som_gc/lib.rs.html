50    pub fn object_start_to_ref(start: Address) -> ObjectReference {
51        // Safety: start is the allocation result, and it should not be zero with an offset.
52        unsafe { ObjectReference::from_raw_address_unchecked(start + object_model::OBJECT_REF_OFFSET) }
53    }
