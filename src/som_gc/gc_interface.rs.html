73    /// Initialize the GCInterface. Internally inits MMTk and fetches everything needed to actually communicate with the GC.
74    pub fn init<'a>(heap_size: usize, vm_callbacks: MMTKtoVMCallbacks) -> &'a mut Self {
75        let (mutator_thread, mutator) = Self::init_mmtk(heap_size);
76        #[cfg(feature = "marksweep")]
77        let default_allocator = Self::get_default_allocator::<FreeListAllocator<SOMVM>>(mutator.as_ref());
78        #[cfg(feature = "semispace")]
79        let default_allocator = Self::get_default_allocator::<BumpAllocator<SOMVM>>(mutator.as_ref());
80
81        let self_ = Box::new(Self {
82            mutator_thread,
83            mutator,
84            is_collecting: false,
85            default_allocator,
86            #[cfg(feature = "semispace")]
87            alloc_bump_ptr: BumpPointer::default(),
88            start_the_world_count: 0,
89            total_gc_time: Duration::new(0, 0),
90        });
91
92        let gc_interface_ptr = Box::leak(self_);
93
94        unsafe {
95            // in the context of tests, this function gets invoked many times, so they can have already been initialized.
96            // TODO: which makes me realize that this function's structure is subpar. Why do we return a NEW GCInterface at all, then?
97            // The universe should likely use a reference to the OnceCell, or something... That'd be better.
98
99            if MUTATOR_WRAPPER.get().is_none() {
100                // very unsafe, very ugly: we duplicate a mutable reference to the GC interface ptr. need to avoid by implementing above idea
101                let dup_ptr = &mut *(gc_interface_ptr as *mut GCInterface);
102                MUTATOR_WRAPPER.set(dup_ptr).unwrap_or_else(|_| panic!("couldn't set mutator wrapper?"));
103            }
104
105            if MMTK_TO_VM_INTERFACE.get().is_none() {
106                MMTK_TO_VM_INTERFACE.get_or_init(|| vm_callbacks);
107            }
108        }
109
110        gc_interface_ptr
111    }

113    /// Initialize MMTk, and get from it all the info we need to initialize our interface
114    fn init_mmtk(heap_size: usize) -> (VMMutatorThread, Box<Mutator<SOMVM>>) {
115        let builder: MMTKBuilder = {
116            let mut builder = MMTKBuilder::new();
117
118            let heap_success = mmtk_set_fixed_heap_size(&mut builder, heap_size);
119            assert!(heap_success, "Couldn't set MMTk fixed heap size");
120
121            if cfg!(feature = "marksweep") {
122                assert!(builder.set_option("plan", "MarkSweep"));
123            } else if cfg!(feature = "semispace") {
124                assert!(builder.set_option("plan", "SemiSpace"));
125            } else {
126                panic!("No GC plan set!")
127            }
128
129            #[cfg(feature = "stress_test")]
130            assert!(builder.set_option("stress_factor", "4000000"));
131
132            builder
133        };
134
135        if MMTK_SINGLETON.get().is_none() {
136            MMTK_SINGLETON
137                .set({
138                    let mmtk = mmtk::memory_manager::mmtk_init::<SOMVM>(&builder);
139                    *mmtk
140                })
141                .unwrap_or_else(|_| panic!("couldn't set the MMTk singleton"));
142
143            mmtk_initialize_collection(VMThread(OpaquePointer::UNINITIALIZED));
144        }
145
146        let tls = VMMutatorThread(VMThread(OpaquePointer::UNINITIALIZED));
147        let mutator = mmtk_bind_mutator(tls);
148
149        (tls, mutator)
150    }

152    fn get_default_allocator<T>(mutator: &Mutator<SOMVM>) -> *mut T
153    where
154        T: Allocator<SOMVM>,
155    {
156        let selector = memory_manager::get_allocator_mapping(MMTK_SINGLETON.get().unwrap(), AllocationSemantics::Default);
157        let default_allocator_offset = Mutator::<SOMVM>::get_allocator_base_offset(selector);
158
159        // At run time: allocate with the default semantics without resolving allocator
160        let default_allocator: *mut T = {
161            let mutator_addr = Address::from_ref(mutator);
162            unsafe {
163                let allocator_ptr = mutator_addr + default_allocator_offset;
164                allocator_ptr.as_mut_ref()
165            }
166        };
167
168        default_allocator
169    }

197    /// Block the main thread to perform GC.
198    pub(crate) fn block_for_gc(&mut self, _tls: VMMutatorThread) {
199        debug!("block_for_gc: stopping the world!");
200        self.is_collecting = true;
201
202        let (is_world_stopped, cvar) = &*WORLD_LOCK;
203        {
204            let mut lock = is_world_stopped.lock().unwrap();
205            *lock = true;
206        }
207
208        let time_pre_gc = Instant::now();
209
210        let result = cvar.wait_timeout_while(is_world_stopped.lock().unwrap(), Duration::from_secs(15), |pending| *pending).unwrap();
211        if result.1.timed_out() {
212            panic!("GC timed out: highly likely to be a crash in a GC thread.")
213        }
214
215        debug!("block_for_gc: world no longer stopped.");
216        self.is_collecting = false;
217        self.total_gc_time += Instant::now() - time_pre_gc;
218    }

233    pub(crate) fn resume_mutators(&mut self) {
234        debug!("resuming mutators.");
235        self.start_the_world_count += 1;
236
237        let (is_world_stopped, cvar) = &*WORLD_LOCK;
238        let mut pending = is_world_stopped.lock().unwrap();
239        *pending = false;
240        cvar.notify_one();
241    }

320    fn alloc_with_marker<T: HasTypeInfoForGC>(&mut self, obj: T, alloc_origin_marker: Option<AllocSiteMarker>) -> Gc<T> {

324    /// Allocates a type, but with a given size.
325    fn alloc_with_size<T: HasTypeInfoForGC>(&mut self, obj: T, size: usize, alloc_origin_marker: Option<AllocSiteMarker>) -> Gc<T> {
326        debug_assert!(size >= MIN_OBJECT_SIZE);
327
328        // adding VM header size (type info) to amount we allocate
329        let header_addr = self.request_bytes(size + OBJECT_REF_OFFSET, alloc_origin_marker);
330
331        debug_assert!(!header_addr.is_zero());
332        let obj_addr = SOMVM::object_start_to_ref(header_addr);
333
334        // AFAIK, this is not needed.
335        // mmtk_post_alloc(mutator, SOMVM::object_start_to_ref(addr), size, GC_SEMANTICS);
336
337        unsafe {
338            *header_addr.as_mut_ref() = T::get_magic_gc_id();
339            *(obj_addr.to_raw_address().as_mut_ref()) = obj;
340        }
341
342        Gc::from(obj_addr.to_raw_address())
343    }

345    /// Allocates a slice that only contains values that ARE NOT pointers.
346    /// Allocating a Vec<i32> is fine, allocating a Vec<Value> is fine if they're all Integer values.
347    /// Not unforced by the Rust type system atm, but we could make some nice traits for this. Just afraid that this would add unnecessary complexity.
348    fn alloc_safe_slice<T: SupportedSliceType + std::fmt::Debug>(&mut self, obj: &[T]) -> GcSlice<T> {
349        self.alloc_safe_slice_with_marker(obj, None)

359    }

361    /// Deprecated because too likely to be unsafe: GC triggered when allocating a slice makes the

380    fn write_slice_to_addr<T: SupportedSliceType + std::fmt::Debug>(&mut self, slice_header_addr: Address, obj: &[T]) -> GcSlice<T> {
381        let len_addr = SOMVM::object_start_to_ref(slice_header_addr);
382        let obj_addr = len_addr.to_raw_address().add(size_of::<usize>());
383
384        unsafe {
385            *slice_header_addr.as_mut_ref() = T::get_magic_gc_slice_id();
386            *len_addr.to_raw_address().as_mut_ref() = obj.len();
387            std::ptr::copy_nonoverlapping(obj.as_ptr(), obj_addr.as_mut_ref(), obj.len());
388        }
389
390        GcSlice::new(len_addr.to_raw_address())
391    }

453    /// TODO doc + should likely deduce the size from the type
454    fn request_memory_for_type<T: HasTypeInfoForGC>(&mut self, type_size: usize, alloc_origin_marker: Option<AllocSiteMarker>) -> Gc<T> {
455        let mut bytes = self.request_bytes(type_size + OBJECT_REF_OFFSET, alloc_origin_marker);
456        unsafe {
457            *bytes.as_mut_ref::<u8>() = T::get_magic_gc_id();
458            bytes += OBJECT_REF_OFFSET;
459            bytes.into()
460        }
461    }

463    fn request_bytes_for_slice(&mut self, slice_size: usize, alloc_origin_marker: Option<AllocSiteMarker>) -> Address {
464        let mut size = {
465            match slice_size {
466                v if v < MIN_OBJECT_SIZE => MIN_OBJECT_SIZE,
467                v => v,
468            }
469        };
470
471        size += std::mem::size_of::<usize>(); // size stored at the start
472
473        // slices can be big enough to warrant using large object storage.
474        let header_addr = {
475            match size <= crate::mmtk().get_plan().constraints().max_non_los_default_alloc_bytes {
476                true => self.request_bytes(size + OBJECT_REF_OFFSET, alloc_origin_marker),
477                false => self.request_bytes_los(size + OBJECT_REF_OFFSET, alloc_origin_marker),
478            }
479        };
480
481        header_addr
482    }
