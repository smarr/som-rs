42fn load_file(interpreter: &mut Interpreter, universe: &mut Universe) -> Result<Option<Gc<String>>, Error> {
43    pop_args_from_stack!(interpreter, _a => Value, path => StringLike);
44    let path = path.as_str(|sym| universe.lookup_symbol(sym));
45
46    let Ok(value) = fs::read_to_string(&*path) else {
47        return Ok(None);
48    };
49
50    Ok(Some(universe.gc_interface.alloc(value)))
51}

141fn print_stack_trace(interpreter: &mut Interpreter, _: &mut Universe) -> Result<bool, Error> {
142    pop_args_from_stack!(interpreter, _a => Value);
143
144    let frame_stack = {
145        let mut frame_stack = vec![];
146        let mut current_frame = interpreter.get_current_frame();
147        while !current_frame.is_empty() {
148            frame_stack.push(current_frame.clone());
149            current_frame = current_frame.prev_frame.clone();
150        }
151        frame_stack
152    };
153
154    println!("Stack trace:");
155    for (frame_idx, frame) in frame_stack.iter().enumerate() {
156        let class = frame.get_method_holder();
157        println!(
158            "\t{}: {}>>#{} @bi: {}",
159            frame_idx,
160            class.name(),
161            frame.current_context.signature(),
162            frame.bytecode_idx
163        );
164    }
165    println!("----------------");
166
167    Ok(true)
168}

175fn gc_stats(interp: &mut Interpreter, universe: &mut Universe) -> Result<VecValue, Error> {
176    pop_args_from_stack!(interp, _a => Value);
177
178    let gc_interface = &mut universe.gc_interface;
179
180    let total_gc = gc_interface.get_nbr_collections();
181    let total_gc_time = gc_interface.alloc(BigInt::from(gc_interface.get_total_gc_time()));
182    let total_bytes_bigint = gc_interface.alloc(BigInt::from(gc_interface.get_used_bytes()));
183
184    Ok(VecValue(universe.gc_interface.alloc_slice(&[
185        Value::Integer(total_gc as i32),
186        Value::BigInteger(total_gc_time),
187        Value::BigInteger(total_bytes_bigint),
188    ])))
189}
