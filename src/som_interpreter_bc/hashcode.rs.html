11impl Hash for Value {
12    fn hash<H: Hasher>(&self, hasher: &mut H) {
13        if self.is_nil() {
14            hasher.write(b"#nil#");
15        } else if let Some(value) = self.as_boolean() {
16            hasher.write(b"#bool#");
17            value.hash(hasher);
18        } else if let Some(value) = self.as_integer() {
19            hasher.write(b"#int#");
20            value.hash(hasher);
21        } else if let Some(value) = self.as_big_integer::<Gc<BigInt>>() {
22            hasher.write(b"#bigint#");
23            value.hash(hasher);
24        } else if let Some(value) = self.as_double() {
25            hasher.write(b"#double#");
26            let raw_bytes: &[u8] = unsafe { std::slice::from_raw_parts((&value as *const f64) as *const u8, std::mem::size_of::<f64>()) };
27            hasher.write(raw_bytes);
28        } else if let Some(value) = self.as_symbol() {
29            hasher.write(b"#sym#");
30            value.hash(hasher);
31        } else if let Some(value) = self.as_string::<Gc<String>>() {
32            hasher.write(b"#string#");
33            value.hash(hasher);
34        } else if let Some(value) = self.as_array() {
35            hasher.write(b"#arr#");
36            value.0.iter().for_each(|elem| elem.hash(hasher));
37        } else if let Some(value) = self.as_block() {
38            hasher.write(b"#blk#");
39            value.hash(hasher);
40        } else if let Some(value) = self.as_class() {
41            hasher.write(b"#cls#");
42            value.deref().hash(hasher);
43        } else if let Some(instance) = self.as_instance() {
44            hasher.write(b"#inst#");
45            instance.class.hash(hasher);
46            for i in 0..instance.class.fields.len() {
47                Instance::lookup_field(&instance, i).hash(hasher)
48            }
49        } else if let Some(value) = self.as_invokable() {
50            hasher.write(b"#mthd#");
51            value.hash(hasher);
52        } else {
53            panic!("Unexpected Value variant encountered!");
54        }
55    }
56}

58impl Hash for Class {
59    fn hash<H: Hasher>(&self, hasher: &mut H) {
60        self.name.hash(hasher);
61        self.fields.iter().for_each(|value| {
62            value.hash(hasher);
63        });
64    }
65}

67impl Hash for Block {
68    fn hash<H: Hasher>(&self, hasher: &mut H) {
69        let blk_info = &self.blk_info;
70        blk_info.get_env().literals.iter().for_each(|it| it.hash(hasher));
71        blk_info.get_env().nbr_locals.hash(hasher);
72        // self.blk_info.locals.iter().for_each(|it| it.hash(hasher));
73        blk_info.get_env().nbr_params.hash(hasher);
74        blk_info.get_env().body.iter().for_each(|it| it.hash(hasher));
75    }
76}

78impl Hash for Method {
79    fn hash<H: Hasher>(&self, hasher: &mut H) {
80        self.holder().hash(hasher);
81        hasher.write(b">>");
82        self.signature().hash(hasher);
83    }
