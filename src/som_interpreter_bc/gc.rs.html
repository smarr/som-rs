105/// Visits a value, via a specialized `SOMSlot` for value types.
106/// # Safety
107/// Values passed to this function MUST live on the GC heap, or the pointer generated from the reference will be invalid.
108unsafe fn visit_value<'a>(val: &Value, slot_visitor: &'a mut (dyn SlotVisitor<SOMSlot> + 'a)) {
109    if val.is_ptr_type() {
110        if let Some(slice) = val.as_array() {
111            // large object storage means no copying needed, but we still check the values stored
112            if slice.get_true_size() >= 65535 {
113                for val in slice.iter() {
114                    visit_value(val, slot_visitor)
115                }
116                return;
117            }
118        }
119        slot_visitor.visit_slot(SOMSlot::from(val.as_mut_ptr()))
120    }
121}
122
123/// Visits a value and potentially adds a slot made out of it to an array.
124/// # Safety
125/// Same as `visit_value`.
126unsafe fn visit_value_maybe_process(val: &Value, to_process: &mut Vec<SOMSlot>) {
127    if val.is_ptr_type() {
128        if let Some(slice) = val.as_array() {
129            // large object storage means no copying needed, but we still check the values stored
130            if slice.get_true_size() >= 65535 {
131                for val2 in slice.iter() {
132                    visit_value_maybe_process(val2, to_process);
133                }
134                return;
135            }
136        }
137        to_process.push(SOMSlot::from(val.as_mut_ptr()))
138    }
139}

141pub fn visit_literal<'a>(lit: &Literal, slot_visitor: &'a mut (dyn SlotVisitor<SOMSlot> + 'a)) {
142    match lit {
143        Literal::Block(blk) => slot_visitor.visit_slot(SOMSlot::from(blk)),
144        Literal::String(str) => slot_visitor.visit_slot(SOMSlot::from(str)),
145        Literal::BigInteger(bigint) => slot_visitor.visit_slot(SOMSlot::from(bigint)),
146        Literal::Array(arr) => slot_visitor.visit_slot(SOMSlot::from(arr)),
147        _ => {}
148    }
149}

151pub fn scan_object<'a>(object: ObjectReference, slot_visitor: &'a mut (dyn SlotVisitor<SOMSlot> + 'a)) {
152    unsafe {
153        let gc_id: &BCObjMagicId = VMObjectModel::ref_to_header(object).as_ref();
154
155        trace!("entering scan_object (type: {:?})", gc_id);
156
157        match gc_id {
158            BCObjMagicId::Frame => {
159                let frame: &Frame = object.to_raw_address().as_ref();
160
161                if !frame.prev_frame.is_empty() {
162                    slot_visitor.visit_slot(SOMSlot::from(&frame.prev_frame));
163                }
164
165                slot_visitor.visit_slot(SOMSlot::from(&frame.current_context));
166
167                for i in 0..frame.get_nbr_locals() {
168                    let val: &Value = frame.lookup_local(i);
169                    visit_value(val, slot_visitor)
170                }
171
172                for i in 0..frame.get_nbr_args() {
173                    let val: &Value = frame.lookup_argument(i);
174                    visit_value(val, slot_visitor)
175                }
176
177                let stack_iter = FrameStackIter::from(frame);
178                for stack_item in stack_iter.into_iter() {
179                    visit_value(stack_item, slot_visitor);
180                }
181            }
182            BCObjMagicId::Method => {
183                let method: &Method = object.to_raw_address().as_ref();
184
185                match method {
186                    Method::Defined(method) => {
187                        slot_visitor.visit_slot(SOMSlot::from(&method.base_method_info.holder));
188
189                        for cache_entry in method.inline_cache.iter().flatten() {
190                            match cache_entry {
191                                CacheEntry::Send(cls_ptr, method_ptr) => {
192                                    slot_visitor.visit_slot(SOMSlot::from(cls_ptr));
193                                    slot_visitor.visit_slot(SOMSlot::from(method_ptr));
194                                }
195                                CacheEntry::Global(val) => {
196                                    visit_value(val, slot_visitor);
197                                }
198                            }
199                        }
200
201                        for lit in &method.literals {
202                            visit_literal(lit, slot_visitor)
203                        }
204                    }
205                    Method::Primitive(_, met_info)
206                    | Method::TrivialGlobal(_, met_info)
207                    | Method::TrivialGetter(_, met_info)
208                    | Method::TrivialSetter(_, met_info) => {
209                        slot_visitor.visit_slot(SOMSlot::from(&met_info.holder));
210                    }
211                    Method::TrivialLiteral(trivial_lit, met_info) => {
212                        visit_literal(&trivial_lit.literal, slot_visitor);
213                        slot_visitor.visit_slot(SOMSlot::from(&met_info.holder));
214                    }
215                }
216            }
217            BCObjMagicId::Class => {
218                let class: &Class = object.to_raw_address().as_ref();
219
220                slot_visitor.visit_slot(SOMSlot::from(&class.class));
221
222                if class.super_class.is_some() {
223                    slot_visitor.visit_slot(SOMSlot::from(class.super_class.as_ref().unwrap()));
224                }
225
226                for (_, method_ref) in class.methods.iter() {
227                    slot_visitor.visit_slot(SOMSlot::from(method_ref))
228                }
229
230                for field_ref in class.fields.iter() {
231                    visit_value(field_ref, slot_visitor)
232                }
233            }
234            BCObjMagicId::Block => {
235                let block: &Block = object.to_raw_address().as_ref();
236
237                if let Some(frame) = block.frame.as_ref() {
238                    slot_visitor.visit_slot(SOMSlot::from(frame));
239                }
240
241                slot_visitor.visit_slot(SOMSlot::from(&block.blk_info));
242            }
243            BCObjMagicId::Instance => {
244                let instance: &Instance = object.to_raw_address().as_ref();
245                slot_visitor.visit_slot(SOMSlot::from(&instance.class));
246
247                let instance_as_gc: Gc<Instance> = object.to_raw_address().into();
248                for i in 0..instance.class().get_nbr_fields() {
249                    let val: &Value = Instance::lookup_field(&instance_as_gc, i);
250                    visit_value(val, slot_visitor)
251                }
252            }
253            BCObjMagicId::ArrayVal => {
254                let arr: GcSlice<Value> = GcSlice::from(object.to_raw_address());
255                for val in arr.iter() {
256                    visit_value(val, slot_visitor)
257                }
258            }
259            BCObjMagicId::ArrayLiteral => {
260                let literal_vec: GcSlice<Literal> = GcSlice::from(object.to_raw_address());
261                for lit in literal_vec.iter() {
262                    visit_literal(lit, slot_visitor)
263                }
264            }
265            BCObjMagicId::String | BCObjMagicId::BigInt => {
266                // leaf nodes: no children.
267            }
268        }
269    }
270}

272fn get_roots_in_mutator_thread(_mutator: &mut Mutator<SOMVM>) -> Vec<SOMSlot> {
273    debug!("calling scan_roots_in_mutator_thread");
274    unsafe {
275        let mut to_process: Vec<SOMSlot> = vec![];
276
277        assert!(
278            !(*UNIVERSE_RAW_PTR_CONST.as_ptr()).is_null() && !(*INTERPRETER_RAW_PTR_CONST.as_ptr()).is_null(),
279            "GC triggered while the system wasn't finished initializing."
280        );
281
282        // walk the frame list.
283        let current_frame_addr = &*(**INTERPRETER_RAW_PTR_CONST.as_ptr()).current_frame.get();
284        debug!(
285            "scanning root: current_frame (method: {})",
286            current_frame_addr.current_context.get_env().base_method_info.signature
287        );
288        to_process.push(SOMSlot::from(current_frame_addr));
289
290        let frame_method_root = &(**INTERPRETER_RAW_PTR_CONST.as_ptr()).frame_method_root;
291        if !frame_method_root.is_empty() {
292            to_process.push(SOMSlot::from(frame_method_root));
293        }
294
295        let frame_args_root = &(**INTERPRETER_RAW_PTR_CONST.as_ptr()).frame_args_root;
296        if let Some(frame_args) = frame_args_root {
297            for arg in frame_args {
298                visit_value_maybe_process(arg, &mut to_process);
299            }
300        }
301
302        // walk globals (includes core classes)
303        debug!("scanning roots: globals");
304        for (_name, val) in (**UNIVERSE_RAW_PTR_CONST.as_ptr()).globals.iter_mut() {
305            visit_value_maybe_process(val, &mut to_process);
306        }
307
308        // we update the core classes in their class also though, to properly move them
309        (**UNIVERSE_RAW_PTR_CONST.as_ptr()).core.iter().for_each(|(_, cls_ptr)| to_process.push(SOMSlot::from(cls_ptr)));
310
311        debug!("scanning roots: finished");
312        to_process
313    }
314}

316fn get_object_size(object: ObjectReference) -> usize {
317    let gc_id: &BCObjMagicId = unsafe { VMObjectModel::ref_to_header(object).as_ref() };
318
319    let obj_size = {
320        match gc_id {
321            BCObjMagicId::String => size_of::<String>(),
322            BCObjMagicId::BigInt => size_of::<BigInt>(),
323            BCObjMagicId::ArrayLiteral => {
324                let literals: GcSlice<Literal> = GcSlice::from(object.to_raw_address());
325                literals.get_true_size()
326            }
327            BCObjMagicId::Frame => unsafe {
328                let frame: &Frame = object.to_raw_address().as_ref();
329                Frame::get_true_size(frame.get_max_stack_size(), frame.get_nbr_args(), frame.get_nbr_locals())
330            },
331            BCObjMagicId::ArrayVal => {
332                let values: GcSlice<Value> = GcSlice::from(object.to_raw_address());
333                values.get_true_size()
334            }
335            BCObjMagicId::Method => size_of::<Method>(),
336            BCObjMagicId::Block => size_of::<Block>(),
337            BCObjMagicId::Class => size_of::<Class>(),
338            BCObjMagicId::Instance => unsafe {
339                let instance: &Instance = object.to_raw_address().as_ref();
340                size_of::<Instance>() + instance.class.fields.len() * size_of::<Value>()
341            },
342        }
343    };
344
345    // debug!("get object size invoked ({:?}), and returning {}", gc_id, obj_size);
346
347    obj_size
348}
