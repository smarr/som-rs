54/// Calculates the maximum stack size possible. For each frame, this allows us to allocate a stack of precisely the maximum possible size it needs.
55/// TODO opt: it's possible our estimate is overly conservative. Reducing the max stack size reduces time spent allocating, and could maybe be a worthwhile optimization.
56pub(crate) fn get_max_stack_size(body: &[Bytecode], interner: &Interner) -> u8 {
57    let mut abstract_stack_size = 0;
58    let mut max_stack_size_observed: u8 = 0;
59
60    for bc in body {
61        match bc {
62            Bytecode::Dup
63            | Bytecode::Dup2
64            | Bytecode::PushLocal(..)
65            | Bytecode::PushNonLocal(..)
66            | Bytecode::PushArg(..)
67            | Bytecode::PushNonLocalArg(..)
68            | Bytecode::PushField(..)
69            | Bytecode::PushBlock(..)
70            | Bytecode::PushConstant(..)
71            | Bytecode::PushGlobal(..)
72            | Bytecode::Push0
73            | Bytecode::Push1
74            | Bytecode::PushNil
75            | Bytecode::PushSelf => {
76                abstract_stack_size += 1;
77                if abstract_stack_size > max_stack_size_observed {
78                    max_stack_size_observed = abstract_stack_size
79                }
80            }
81            Bytecode::Pop
82            | Bytecode::PopLocal(..)
83            | Bytecode::PopArg(..)
84            | Bytecode::PopField(..)
85            | Bytecode::JumpOnTruePop(..)
86            | Bytecode::JumpOnFalsePop(..) => abstract_stack_size -= 1,
87            Bytecode::Send1(_) => {}
88            Bytecode::Send2(_) => abstract_stack_size -= 1, // number of arguments (they all get popped) + 1 for the result
89            Bytecode::Send3(_) => abstract_stack_size -= 2,
90            Bytecode::SendN(symbol) | Bytecode::SuperSend(symbol) => {
91                let nb_params = {
92                    let uninterned = interner.lookup(*symbol);
93                    match uninterned.chars().next() {
94                        Some(ch) if !ch.is_alphabetic() => 1,
95                        _ => uninterned.chars().filter(|ch| *ch == ':').count() as u8,
96                    }
97                };
98
99                if nb_params > 0 {
100                    abstract_stack_size -= nb_params - 1
101                }
102            }
103            Bytecode::Inc => {}
104            Bytecode::Dec => {}
105            Bytecode::ReturnSelf => {}
106            Bytecode::ReturnLocal => {}
107            Bytecode::ReturnNonLocal(_) => {}
108            Bytecode::Jump(_) => {}
109            Bytecode::JumpBackward(_) => {}
110            Bytecode::JumpOnTrueTopNil(_) => {}
111            Bytecode::JumpOnFalseTopNil(_) => {}
112            Bytecode::JumpOnNilTopTop(_) => {}
113            Bytecode::JumpOnNotNilTopTop(_) => {}
114            Bytecode::JumpOnNilPop(_) => {}
115            Bytecode::JumpOnNotNilPop(_) => {}
116            Bytecode::JumpIfGreater(_) => {}
117        }
118    }
119
120    // Need to add an extra slot for the hack invoke case
121    max_stack_size_observed + 1
122}

152impl InnerGenCtxt for BlockGenCtxt<'_> {
153    fn as_gen_ctxt(&mut self) -> &mut dyn GenCtxt {
154        self
155    }
156
157    fn push_instr(&mut self, instr: Bytecode) {
158        let body = self.body.get_or_insert_with(Vec::new);
159        body.push(instr);
160    }
161
162    fn pop_instr(&mut self) {
163        self.body.as_mut().unwrap().pop();
164    }
165
166    fn get_instructions(&self) -> &Vec<Bytecode> {
167        self.body.as_ref().unwrap()
168    }
169
170    fn get_literal(&self, idx: usize) -> Option<&Literal> {
171        self.literals.get_index(idx)
172    }
173
174    fn push_literal(&mut self, literal: Literal) -> usize {
175        let (idx, _) = self.literals.insert_full(literal);
176        idx
177    }
178
179    fn remove_literal(&mut self, idx: usize) -> Option<Literal> {
180        self.literals.shift_remove_index(idx)
181    }
182
183    fn get_cur_instr_idx(&self) -> usize {
184        self.body.as_ref().unwrap().iter().len()
185    }
186
187    fn backpatch_jump_to_current(&mut self, idx_to_backpatch: usize) {
188        let jump_offset = self.get_cur_instr_idx() - idx_to_backpatch;
189        self.patch_jump(idx_to_backpatch, jump_offset as u16)
190    }
191
192    fn patch_jump(&mut self, idx_to_patch: usize, new_val: u16) {
193        match self.body.as_mut().unwrap().get_mut(idx_to_patch).unwrap() {
194            Bytecode::Jump(jump_idx)
195            | Bytecode::JumpBackward(jump_idx)
196            | Bytecode::JumpOnTrueTopNil(jump_idx)
197            | Bytecode::JumpOnFalseTopNil(jump_idx)
198            | Bytecode::JumpOnTruePop(jump_idx)
199            | Bytecode::JumpOnFalsePop(jump_idx)
200            | Bytecode::JumpOnNilTopTop(jump_idx)
201            | Bytecode::JumpOnNotNilTopTop(jump_idx)
202            | Bytecode::JumpOnNilPop(jump_idx)
203            | Bytecode::JumpOnNotNilPop(jump_idx)
204            | Bytecode::JumpIfGreater(jump_idx) => *jump_idx = new_val,
205            _ => panic!("Attempting to patch a bytecode non jump"),
206        };
207    }
208
209    fn get_nbr_locals(&self) -> usize {
210        self.locals_nbr
211    }
212
213    fn set_nbr_locals(&mut self, nbr_locals: usize) {
214        self.locals_nbr = nbr_locals;
215    }
216
217    fn remove_dup_popx_pop_sequences(&mut self) {
218        if self.body.is_none() || self.body.as_ref().unwrap().len() < 3 {
219            return;
220        }
221
222        let body = self.body.as_mut().unwrap();
223
224        let mut indices_to_remove: Vec<usize> = vec![];
225
226        for (idx, bytecode_win) in body.windows(3).enumerate() {
227            if matches!(bytecode_win[0], Bytecode::Dup)
228                && matches!(bytecode_win[1], Bytecode::PopField(..) | Bytecode::PopLocal(..) | Bytecode::PopArg(..))
229                && matches!(bytecode_win[2], Bytecode::Pop)
230            {
231                let are_bc_jump_targets = body.iter().enumerate().any(|(maybe_jump_idx, bc)| match bc {
232                    Bytecode::Jump(jump_offset)
233                    | Bytecode::JumpOnTrueTopNil(jump_offset)
234                    | Bytecode::JumpOnFalseTopNil(jump_offset)
235                    | Bytecode::JumpOnTruePop(jump_offset)
236                    | Bytecode::JumpOnFalsePop(jump_offset)
237                    | Bytecode::JumpIfGreater(jump_offset)
238                    | Bytecode::JumpOnNilPop(jump_offset)
239                    | Bytecode::JumpOnNotNilPop(jump_offset)
240                    | Bytecode::JumpOnNilTopTop(jump_offset)
241                    | Bytecode::JumpOnNotNilTopTop(jump_offset) => {
242                        let bc_target_idx = maybe_jump_idx + *jump_offset as usize;
243                        bc_target_idx == idx || bc_target_idx == idx + 2
244                    }
245                    _ => false,
246                });
247
248                if are_bc_jump_targets {
249                    continue;
250                }
251
252                indices_to_remove.push(idx);
253                indices_to_remove.push(idx + 2);
254            }
255        }
256
257        if indices_to_remove.is_empty() {
258            return;
259        }
260
261        let mut jumps_to_patch: Vec<(usize, u16)> = vec![];
262        for (cur_idx, bc) in body.iter().enumerate() {
263            match bc {
264                Bytecode::Jump(jump_offset)
265                | Bytecode::JumpOnTrueTopNil(jump_offset)
266                | Bytecode::JumpOnFalseTopNil(jump_offset)
267                | Bytecode::JumpOnTruePop(jump_offset)
268                | Bytecode::JumpOnFalsePop(jump_offset)
269                | Bytecode::JumpIfGreater(jump_offset) => {
270                    let jump_offset = *jump_offset as usize;
271
272                    if indices_to_remove.contains(&(cur_idx + jump_offset)) {
273                        panic!("should be unreachable");
274                        // let jump_target_in_removes_idx = indices_to_remove
275                        //     .iter()
276                        //     .position(|&v| v == cur_idx + jump_offset)
277                        //     .unwrap();
278                        // indices_to_remove.remove(jump_target_in_removes_idx);
279                        // // indices_to_remove.remove(jump_target_in_removes_idx - 1);
280                        // let to_remove = (jump_target_in_removes_idx,
281                        //                  match jump_target_in_removes_idx % 2 {
282                        //                      0 => jump_target_in_removes_idx + 1,
283                        //                      1 => jump_target_in_removes_idx - 1,
284                        //                      _ => unreachable!()
285                        //                  });
286                        //
287                        // indices_to_remove.retain(|v| *v != to_remove.0 && *v != to_remove.1);
288                        // continue;
289                    }
290
291                    let nbr_to_adjust = indices_to_remove.iter().filter(|&&idx| cur_idx < idx && idx <= cur_idx + jump_offset).count();
292                    jumps_to_patch.push((cur_idx, (jump_offset - nbr_to_adjust) as u16));
293                }
294                Bytecode::JumpBackward(jump_offset) => {
295                    let jump_offset = *jump_offset as usize;
296                    let nbr_to_adjust = indices_to_remove.iter().filter(|&&idx| cur_idx > idx && idx > cur_idx - jump_offset).count();
297                    jumps_to_patch.push((cur_idx, (jump_offset - nbr_to_adjust) as u16));
298                    // It's impossible for a JumpBackward to be generated to point to a duplicated dup/pop/pox sequence, as it stands, and as far as I know.
299                }
300                _ => {}
301            }
302        }
303
304        for (jump_idx, jump_val) in jumps_to_patch {
305            self.patch_jump(jump_idx, jump_val);
306        }
307
308        let mut index = 0;
309        self.body.as_mut().unwrap().retain(|_| {
310            let is_kept = !indices_to_remove.contains(&index);
311            index += 1;
312            is_kept
313        });
314    }
315}

409impl MethodCodegen for ast::Expression {
410    fn codegen(&self, ctxt: &mut dyn InnerGenCtxt, mutator: &mut GCInterface) -> Option<()> {
411        match self {
412            ast::Expression::LocalVarRead(idx) => {
413                ctxt.push_instr(Bytecode::PushLocal(*idx as u8));
414                Some(())
415            }
416            ast::Expression::NonLocalVarRead(up_idx, idx) => {
417                ctxt.push_instr(Bytecode::PushNonLocal(*up_idx as u8, *idx as u8));
418                Some(())
419            }
420            ast::Expression::ArgRead(up_idx, idx) => {
421                match (up_idx, idx) {
422                    (0, 0) => ctxt.push_instr(Bytecode::PushSelf),
423                    (0, _) => ctxt.push_instr(Bytecode::PushArg(*idx as u8)),
424                    _ => ctxt.push_instr(Bytecode::PushNonLocalArg(*up_idx as u8, *idx as u8)),
425                };
426                Some(())
427            }
428            ast::Expression::GlobalRead(name) => {
429                match ctxt.find_field(name) {
430                    Some(idx) => ctxt.push_instr(Bytecode::PushField(idx as u8)),
431                    None => match name.as_str() {
432                        "nil" => ctxt.push_instr(Bytecode::PushNil),
433                        "super" => match ctxt.get_scope() {
434                            0 => ctxt.push_instr(Bytecode::PushSelf),
435                            scope => ctxt.push_instr(Bytecode::PushNonLocalArg(scope as u8, 0)),
436                        },
437                        _ => {
438                            let name = ctxt.intern_symbol(name);
439                            let idx = ctxt.push_literal(Literal::Symbol(name));
440                            ctxt.push_instr(Bytecode::PushGlobal(idx as u8));
441                        }
442                    },
443                }
444
445                Some(())
446            }
447            ast::Expression::LocalVarWrite(_, expr) | ast::Expression::NonLocalVarWrite(_, _, expr) => {
448                expr.codegen(ctxt, mutator)?;
449                ctxt.push_instr(Bytecode::Dup);
450                match self {
451                    ast::Expression::LocalVarWrite(idx, _) => ctxt.push_instr(Bytecode::PopLocal(0, *idx as u8)),
452                    ast::Expression::NonLocalVarWrite(up_idx, idx, _) => ctxt.push_instr(Bytecode::PopLocal(*up_idx as u8, *idx as u8)),
453                    _ => unreachable!(),
454                }
455                Some(())
456            }
457            ast::Expression::GlobalWrite(name, expr) => match ctxt.find_field(name) {
458                Some(idx) => {
459                    expr.codegen(ctxt, mutator)?;
460                    ctxt.push_instr(Bytecode::Dup);
461                    ctxt.push_instr(Bytecode::PopField(idx as u8));
462                    Some(())
463                }
464                None => panic!("couldn't resolve a globalwrite (`{}`) to a field write", name),
465            },
466            ast::Expression::ArgWrite(up_idx, idx, expr) => {
467                expr.codegen(ctxt, mutator)?;
468                ctxt.push_instr(Bytecode::Dup);
469                ctxt.push_instr(Bytecode::PopArg(*up_idx as u8, *idx as u8));
470                Some(())
471            }
472            ast::Expression::Message(message) => {
473                let is_super_call = matches!(&message.receiver, _super if _super == &Expression::GlobalRead(String::from("super")));
474
475                message.receiver.codegen(ctxt, mutator)?;
476
477                #[cfg(not(feature = "inlining-disabled"))]
478                if message.inline_if_possible(ctxt, mutator).is_some() {
479                    return Some(());
480                }
481
482                if (message.signature == "+" || message.signature == "-")
483                    && !is_super_call
484                    && message.values.len() == 1
485                    && message.values.first()? == &Expression::Literal(ast::Literal::Integer(1))
486                {
487                    match message.signature.as_str() {
488                        "+" => ctxt.push_instr(Bytecode::Inc),
489                        "-" => ctxt.push_instr(Bytecode::Dec),
490                        _ => unreachable!(),
491                    };
492                    return Some(());
493                }
494
495                message.values.iter().try_for_each(|value| value.codegen(ctxt, mutator))?;
496
497                let nb_params = match message.signature.chars().nth(0) {
498                    Some(ch) if !ch.is_alphabetic() => 1,
499                    _ => message.signature.chars().filter(|ch| *ch == ':').count(),
500                };
501
502                let sym = ctxt.intern_symbol(message.signature.as_str());
503
504                match is_super_call {
505                    false => match nb_params {
506                        0 => ctxt.push_instr(Bytecode::Send1(sym)),
507                        1 => ctxt.push_instr(Bytecode::Send2(sym)),
508                        2 => ctxt.push_instr(Bytecode::Send3(sym)),
509                        _ => ctxt.push_instr(Bytecode::SendN(sym)),
510                    },
511                    true => ctxt.push_instr(Bytecode::SuperSend(sym)),
512                }
513
514                Some(())
515            }
516            ast::Expression::Exit(expr, scope) => {
517                match scope {
518                    0 => match expr.as_ref() {
519                        Expression::ArgRead(0, 0) => ctxt.push_instr(Bytecode::ReturnSelf),
520                        _ => {
521                            expr.codegen(ctxt, mutator)?;
522                            ctxt.push_instr(Bytecode::ReturnLocal)
523                        }
524                    },
525                    _ => {
526                        expr.codegen(ctxt, mutator)?;
527                        ctxt.push_instr(Bytecode::ReturnNonLocal(*scope as u8));
528                    }
529                };
530
531                Some(())
532            }
533            ast::Expression::Literal(literal) => {
534                fn convert_literal(ctxt: &mut dyn InnerGenCtxt, literal: &ast::Literal, gc_interface: &mut GCInterface) -> Literal {
535                    match literal {
536                        ast::Literal::Symbol(val) => Literal::Symbol(ctxt.intern_symbol(val.as_str())),
537                        ast::Literal::String(val) => {
538                            // TODO: this whole bit is to avoid redundant literals. previous logic broke with strings being put on the GC heap. is it indicative of a deeper issue with redundant strings?
539                            // it feels a bit bandaid-ey, since I'm not sure where the bug came from exactly.
540                            // it feels like tests should still pass without all this logic, but they don't (see specialized BC PushConstant one), and I'm not *positive* that's normal?
541                            // also NB: this code was a mild speeddown! could be removed, and the PushConst test deactivated/fixed another way, probably. keeping it for now.
542                            let mut i = 0;
543                            loop {
544                                let lit = ctxt.get_literal(i);
545                                match lit {
546                                    None => break Literal::String(gc_interface.alloc(val.clone())), // reached end of literals and no duplicate, we alloc
547                                    Some(str_lit @ Literal::String(str_ptr)) if **str_ptr == *val => break str_lit.clone(),
548                                    _ => {}
549                                }
550                                i += 1;
551                            }
552                        }
553                        ast::Literal::Double(val) => Literal::Double(*val),
554                        ast::Literal::Integer(val) => Literal::Integer(*val),
555                        ast::Literal::BigInteger(big_int_str) => {
556                            // this is to handle a weird corner case where "-2147483648" is considered to be a bigint by the lexer and then parser, when it's in fact just barely in i32 range
557                            match big_int_str.parse::<i32>() {
558                                Ok(x) => Literal::Integer(x),
559                                _ => Literal::BigInteger(gc_interface.alloc(BigInt::from_str(big_int_str).unwrap())),
560                            }
561                        }
562                        ast::Literal::Array(val) => {
563                            let literals: GcSlice<Literal> = {
564                                let literals_vec: Vec<Literal> = val.iter().map(|val| convert_literal(ctxt, val, gc_interface)).collect();
565                                gc_interface.alloc_slice(literals_vec.as_slice())
566                            };
567                            Literal::Array(literals)
568                        }
569                    }
570                }
571
572                let literal = convert_literal(ctxt, literal, mutator);
573
574                match literal {
575                    Literal::Integer(0) => ctxt.push_instr(Bytecode::Push0),
576                    Literal::Integer(1) => ctxt.push_instr(Bytecode::Push1),
577                    _ => {
578                        let idx = ctxt.push_literal(literal);
579                        ctxt.push_instr(Bytecode::PushConstant(idx as u8))
580                    }
581                }
582
583                Some(())
584            }
585            ast::Expression::Block(val) => {
586                let block = compile_block(ctxt.as_gen_ctxt(), val, mutator)?;
587                let block = Literal::Block(mutator.alloc(block));
588                let idx = ctxt.push_literal(block);
589                ctxt.push_instr(Bytecode::PushBlock(idx as u8));
590                Some(())
591            }
592        }
593    }
594}

596struct ClassGenCtxt<'a> {
597    pub name: String,
598    pub fields: IndexSet<Interned>,
599    pub methods: IndexMap<Interned, Gc<Method>>,
600    pub interner: &'a mut Interner,
601}
602
603impl GenCtxt for ClassGenCtxt<'_> {
604    fn intern_symbol(&mut self, name: &str) -> Interned {
605        self.interner.intern(name)
606    }
607
608    fn get_scope(&self) -> usize {
609        unreachable!("Asking for scope in a class generation context?")
610    }
611
612    fn find_field(&mut self, name: &str) -> Option<usize> {
613        self.fields.iter().position(|f_int| self.interner.lookup(*f_int) == name)
614    }
615
616    fn get_interner(&self) -> &Interner {
617        self.interner
618    }
619}

621fn compile_method(outer: &mut dyn GenCtxt, defn: &ast::MethodDef, gc_interface: &mut GCInterface) -> Option<Method> {
622    /// Only add a ReturnSelf at the end of a method if needed: i.e. there's no existing return, and if there is, that it can't be jumped over.
623    fn should_add_return_self(ctxt: &mut MethodGenCtxt, body: &ast::Body) -> bool {
624        if body.exprs.is_empty() {
625            return true;
626        }
627
628        // going back two BC to skip the POP added after each expr.codegen(...).
629        match ctxt.get_instructions().iter().nth_back(1) {
630            // if the last BC is a return, we check whether it can be skipped over. if so, we add a ReturnSelf
631            Some(Bytecode::ReturnLocal) | Some(Bytecode::ReturnNonLocal(_)) | Some(Bytecode::ReturnSelf) => {
632                let idx_of_pop_before_potential_return_self = ctxt.get_instructions().len() - 1;
633
634                ctxt.get_instructions().iter().enumerate().any(|(bc_idx, bc)| match bc {
635                    Bytecode::Jump(jump_idx)
636                    | Bytecode::JumpOnTrueTopNil(jump_idx)
637                    | Bytecode::JumpOnFalseTopNil(jump_idx)
638                    | Bytecode::JumpOnTruePop(jump_idx)
639                    | Bytecode::JumpOnFalsePop(jump_idx)
640                    | Bytecode::JumpIfGreater(jump_idx) => bc_idx + *jump_idx as usize >= idx_of_pop_before_potential_return_self,
641                    _ => false,
642                })
643            }
644            _ => true,
645        }
646    }
647
648    fn make_trivial_method_if_possible(body: &Vec<Bytecode>, literals: &[Literal], signature: &str, nbr_params: usize) -> Option<Method> {
649        match (body.as_slice(), nbr_params) {
650            ([Bytecode::PushGlobal(x), Bytecode::ReturnLocal], 0) => match literals.get(*x as usize)? {
651                Literal::Symbol(interned) => Some(Method::TrivialGlobal(
652                    TrivialGlobalMethod {
653                        global_name: *interned,
654                        cached_entry: Cell::new(None),
655                    },
656                    BasicMethodInfo::new(String::from(signature), Gc::default()),
657                )),
658                _ => None,
659            },
660            ([Bytecode::PushField(x), Bytecode::ReturnLocal], 0) => Some(Method::TrivialGetter(
661                TrivialGetterMethod { field_idx: *x },
662                BasicMethodInfo::new(String::from(signature), Gc::default()),
663            )),
664            ([Bytecode::PushArg(1), Bytecode::PopField(x), Bytecode::ReturnSelf], 1) => Some(Method::TrivialSetter(
665                TrivialSetterMethod { field_idx: *x },
666                BasicMethodInfo::new(String::from(signature), Gc::default()),
667            )),
668            ([literal_bc, Bytecode::ReturnLocal], 0) => {
669                let maybe_literal = match literal_bc {
670                    Bytecode::PushConstant(x) => literals.get(*x as usize),
671                    Bytecode::Push0 => Some(&Literal::Integer(0)),
672                    Bytecode::Push1 => Some(&Literal::Integer(1)),
673                    // this case breaks, which i'm not sure makes sense. it's pretty much unused in our benchmarks anyway + AST doesn't have an equivalent optim like that, so it's OK.
674                    // Bytecode::PushBlock(x) => literals.get(*x as usize),
675                    _ => None,
676                };
677
678                maybe_literal.map(|lit| {
679                    Method::TrivialLiteral(
680                        TrivialLiteralMethod { literal: lit.clone() },
681                        BasicMethodInfo::new(String::from(signature), Gc::default()),
682                    )
683                })
684            }
685            _ => None,
686        }
687    }
688
689    let mut ctxt = MethodGenCtxt {
690        signature: defn.signature.clone(),
691        inner: BlockGenCtxt {
692            outer,
693            // args: {
694            //     let mut args = IndexSet::new();
695            //     args.insert(String::from("self"));
696            //     args
697            // },
698            // locals: match &defn.body {
699            //     ast::MethodBody::Primitive => IndexSet::new(),
700            //     ast::MethodBody::Body { locals, .. } => locals.iter().cloned().collect(),
701            // },
702            literals: IndexSet::new(),
703            body: None,
704            locals_nbr: {
705                match &defn.body {
706                    MethodBody::Primitive => 0,
707                    MethodBody::Body { locals_nbr, .. } => *locals_nbr,
708                }
709            },
710            args_nbr: {
711                match defn.signature.chars().next().unwrap() {
712                    '~' | '&' | '|' | '*' | '/' | '\\' | '+' | '=' | '>' | '<' | ',' | '@' | '%' | '-' => 2,
713                    _ => defn.signature.chars().filter(|c| *c == ':').count(),
714                }
715            },
716            #[cfg(feature = "frame-debug-info")]
717            debug_info: {
718                match &defn.body {
719                    MethodBody::Primitive => BlockDebugInfo {
720                        parameters: vec![],
721                        locals: vec![],
722                    },
723                    MethodBody::Body { debug_info, .. } => debug_info.clone(),
724                }
725            },
726        },
727    };
728
729    // match &defn.kind {
730    //     ast::MethodKind::Unary => {}
731    //     ast::MethodKind::Positional { parameters } => {
732    //         for param in parameters {
733    //             ctxt.push_arg(param.clone());
734    //         }
735    //     }
736    //     ast::MethodKind::Operator { rhs } => {
737    //         ctxt.push_arg(rhs.clone());
738    //     }
739    // }
740
741    match &defn.body {
742        ast::MethodBody::Primitive => {}
743        ast::MethodBody::Body { body, .. } => {
744            for expr in &body.exprs {
745                expr.codegen(&mut ctxt, gc_interface)?;
746                ctxt.push_instr(Bytecode::Pop);
747            }
748
749            if should_add_return_self(&mut ctxt, body) {
750                ctxt.push_instr(Bytecode::ReturnSelf);
751            } else {
752                ctxt.pop_instr(); // we can otherwise remove the then-redundant final POP (since it's after an unavoidable "Return" type bytecode.
753            };
754
755            ctxt.remove_dup_popx_pop_sequences();
756        }
757    }
758
759    let method = {
760        match &defn.body {
761            ast::MethodBody::Primitive => Method::Primitive(&*UNIMPLEM_PRIMITIVE, BasicMethodInfo::new(String::from(""), Gc::default())),
762            ast::MethodBody::Body { .. } => {
763                // let locals = std::mem::take(&mut ctxt.inner.locals);
764                let nbr_locals = ctxt.inner.locals_nbr;
765                let body = ctxt.inner.body.clone().unwrap_or_default();
766                let literals: Vec<Literal> = ctxt.inner.literals.clone().into_iter().collect();
767                let signature = ctxt.signature.clone();
768                let max_stack_size = get_max_stack_size(&body, ctxt.get_interner());
769                let nbr_params = {
770                    match ctxt.signature.chars().next() {
771                        Some(ch) if !ch.is_alphabetic() => 1,
772                        _ => ctxt.signature.chars().filter(|ch| *ch == ':').count(),
773                    }
774                };
775
776                if let Some(trivial_method) = make_trivial_method_if_possible(&body, &literals, &signature, nbr_params) {
777                    trivial_method
778                } else {
779                    let inline_cache = vec![None; body.len()];
780                    #[cfg(feature = "frame-debug-info")]
781                    let dbg_info = ctxt.inner.debug_info;
782
783                    Method::Defined(MethodInfo {
784                        base_method_info: BasicMethodInfo::new(signature, Gc::default()),
785                        body,
786                        nbr_locals,
787                        nbr_params,
788                        literals,
789                        inline_cache,
790                        max_stack_size,
791                        #[cfg(feature = "frame-debug-info")]
792                        block_debug_info: dbg_info,
793                    })
794                }
795            }
796        }
797    };
798
799    // println!("(method) compiled '{}' !", defn.signature);
800
801    Some(method)
802}

804fn compile_block(outer: &mut dyn GenCtxt, defn: &ast::Block, gc_interface: &mut GCInterface) -> Option<Block> {
805    // println!("(system) compiling block ...");
806
807    let mut ctxt = BlockGenCtxt {
808        outer,
809        args_nbr: defn.nbr_params,
810        locals_nbr: defn.nbr_locals,
811        // dbg_info: defn.dbg_info,
812        literals: IndexSet::new(),
813        body: None,
814        #[cfg(feature = "frame-debug-info")]
815        debug_info: defn.dbg_info.clone(),
816    };
817
818    let splitted = defn.body.exprs.split_last();
819    if let Some((last, rest)) = splitted {
820        for expr in rest {
821            expr.codegen(&mut ctxt, gc_interface)?;
822            ctxt.push_instr(Bytecode::Pop);
823        }
824        last.codegen(&mut ctxt, gc_interface)?;
825        ctxt.push_instr(Bytecode::ReturnLocal);
826    }
827    ctxt.remove_dup_popx_pop_sequences();
828
829    if ctxt.body.is_none() {
830        ctxt.push_instr(Bytecode::PushNil);
831        ctxt.push_instr(Bytecode::ReturnLocal);
832    }
833
834    let frame = None;
835    // let locals = {
836    // let locals = std::mem::take(&mut ctxt.locals);
837    // locals
838    //     .into_iter()
839    //     .map(|name| ctxt.intern_symbol(&name))
840    //     .collect()
841    // };
842    let literals: Vec<Literal> = ctxt.literals.clone().into_iter().collect();
843    let signature = String::from("--block--");
844    let body = ctxt.body.clone().unwrap_or_default();
845    let nbr_locals = ctxt.locals_nbr;
846    let nbr_params = ctxt.args_nbr;
847    let inline_cache = vec![None; body.len()];
848    let max_stack_size = get_max_stack_size(&body, ctxt.get_interner());
849
850    let block = Block {
851        frame,
852        blk_info: gc_interface.alloc(Method::Defined(MethodInfo {
853            base_method_info: BasicMethodInfo::new(signature, Gc::default()),
854            nbr_locals,
855            literals,
856            body,
857            nbr_params,
858            inline_cache,
859            max_stack_size,
860            #[cfg(feature = "frame-debug-info")]
861            block_debug_info: ctxt.debug_info,
862        })),
863    };
864
865    // println!("(system) compiled block !");
866
867    Some(block)
868}

870pub fn compile_class(
871    interner: &mut Interner,
872    defn: &ast::ClassDef,
873    super_class: Option<&Gc<Class>>,
874    gc_interface: &mut GCInterface,
875) -> Option<Gc<Class>> {
876    let mut locals = IndexSet::new();
877
878    fn collect_static_locals(class: &Gc<Class>, locals: &mut IndexSet<Interned>) {
879        if let Some(class) = class.super_class() {
880            collect_static_locals(&class, locals);
881        }
882        locals.extend(&class.field_names);
883    }
884
885    if let Some(super_class) = super_class {
886        collect_static_locals(&super_class.class(), &mut locals);
887    }
888
889    locals.extend(defn.static_locals.iter().map(|name| interner.intern(name.as_str())));
890
891    let mut static_class_ctxt = ClassGenCtxt {
892        name: format!("{} class", defn.name),
893        fields: locals,
894        methods: IndexMap::new(),
895        interner,
896    };
897
898    let static_class = Class {
899        name: static_class_ctxt.name.clone(),
900        class: Gc::default(),
901        super_class: None,
902        fields: vec![],
903        field_names: vec![],
904        methods: IndexMap::new(),
905        is_static: true,
906    };
907
908    let static_class_gc_ptr = gc_interface.alloc_with_marker(static_class, Some(AllocSiteMarker::Class));
909
910    for method in &defn.static_methods {
911        let signature = static_class_ctxt.interner.intern(method.signature.as_str());
912        let mut method = compile_method(&mut static_class_ctxt, method, gc_interface)?;
913        method.set_holder(&static_class_gc_ptr);
914        static_class_ctxt.methods.insert(signature, gc_interface.alloc_with_marker(method, Some(AllocSiteMarker::Method)));
915    }
916
917    if let Some(primitives) = primitives::get_class_primitives(&defn.name) {
918        for &(signature, primitive, _warning) in primitives {
919            //let symbol = static_class_ctxt.interner.intern(signature);
920            //if warning && !static_class_ctxt.methods.contains_key(&symbol) {
921            //    eprintln!("Warning: Primitive '{}' is not in class definition for class '{}'", signature, defn.name);
922            //}
923
924            let method = Method::Primitive(primitive, BasicMethodInfo::new(String::from(signature), static_class_gc_ptr.clone()));
925
926            let signature = static_class_ctxt.interner.intern(signature);
927            static_class_ctxt.methods.insert(signature, gc_interface.alloc_with_marker(method, Some(AllocSiteMarker::Class)));
928        }
929    }
930
931    let mut static_class_mut = static_class_gc_ptr.clone(); // todo couldn't we have done that before
932    static_class_mut.fields = vec![Value::NIL; static_class_ctxt.fields.len()];
933    static_class_mut.field_names = static_class_ctxt.fields.into_iter().collect();
934    static_class_mut.methods = static_class_ctxt.methods;
935    // drop(static_class_mut);
936
937    // for method in static_class.borrow().methods.values() {
938    //     println!("{}", method);
939    // }
940
941    let mut locals = IndexSet::new();
942
943    fn collect_instance_locals(class: &Gc<Class>, locals: &mut IndexSet<Interned>) {
944        if let Some(class) = class.super_class() {
945            collect_instance_locals(&class, locals);
946        }
947        locals.extend(&class.field_names);
948    }
949
950    if let Some(super_class) = super_class {
951        collect_instance_locals(super_class, &mut locals);
952    }
953
954    locals.extend(defn.instance_locals.iter().map(|name| interner.intern(name.as_str())));
955
956    let mut instance_class_ctxt = ClassGenCtxt {
957        name: defn.name.clone(),
958        fields: locals,
959        methods: IndexMap::new(),
960        interner,
961    };
962
963    let instance_class = Class {
964        name: instance_class_ctxt.name.clone(),
965        class: static_class_gc_ptr,
966        super_class: None,
967        fields: vec![],
968        field_names: vec![],
969        methods: IndexMap::new(),
970        is_static: false,
971    };
972
973    let instance_class_gc_ptr = gc_interface.alloc_with_marker(instance_class, Some(AllocSiteMarker::Class));
974
975    for method in &defn.instance_methods {
976        let signature = instance_class_ctxt.interner.intern(method.signature.as_str());
977        let mut method = compile_method(&mut instance_class_ctxt, method, gc_interface)?;
978        method.set_holder(&instance_class_gc_ptr);
979        instance_class_ctxt.methods.insert(signature, gc_interface.alloc_with_marker(method, Some(AllocSiteMarker::Method)));
980    }
981
982    if let Some(primitives) = primitives::get_instance_primitives(&defn.name) {
983        for &(signature, primitive, _warning) in primitives {
984            //let symbol = instance_class_ctxt.interner.intern(signature);
985            //if warning && !instance_class_ctxt.methods.contains_key(&symbol) {
986            //    eprintln!("Warning: Primitive '{}' is not in class definition for class '{}'", signature, defn.name);
987            //}
988
989            let method = Method::Primitive(primitive, BasicMethodInfo::new(String::from(signature), instance_class_gc_ptr.clone()));
990            let signature = instance_class_ctxt.interner.intern(signature);
991            instance_class_ctxt.methods.insert(signature, gc_interface.alloc_with_marker(method, Some(AllocSiteMarker::Method)));
992        }
993    }
994
995    let mut instance_class_mut = instance_class_gc_ptr.clone();
996    // instance_class_mut.fields = instance_class_ctxt.fields.into_iter().map(|name| (name, Value::NIL)).collect();
997    instance_class_mut.fields = vec![Value::NIL; instance_class_ctxt.fields.len()];
998    instance_class_mut.field_names = instance_class_ctxt.fields.into_iter().collect();
999    instance_class_mut.methods = instance_class_ctxt.methods;
1000    // drop(instance_class_mut);
1001
1002    // for method in instance_class.borrow().methods.values() {
1003    //     println!("{}", method);
1004    // }
1005
1006    // println!("compiled '{}' !", defn.name);
1007
1008    Some(instance_class_gc_ptr)
