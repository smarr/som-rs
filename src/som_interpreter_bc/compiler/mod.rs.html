23#[derive(Debug, Clone)]
24pub enum Literal {
25    Symbol(Interned),
26    String(Gc<String>),
27    Double(f64),
28    Integer(i32),
29    BigInteger(Gc<BigInt>),
30    Array(GcSlice<Literal>),
31    Block(Gc<Block>),
32}
33
34impl PartialEq for Literal {
35    fn eq(&self, other: &Self) -> bool {
36        match (self, other) {
37            (Literal::Symbol(val1), Literal::Symbol(val2)) => val1.eq(val2),
38            (Literal::String(val1), Literal::String(val2)) => val1.eq(val2),
39            (Literal::Double(val1), Literal::Double(val2)) => val1.eq(val2),
40            (Literal::Integer(val1), Literal::Integer(val2)) => val1.eq(val2),
41            (Literal::BigInteger(val1), Literal::BigInteger(val2)) => val1.eq(val2),
42            (Literal::Array(val1), Literal::Array(val2)) => val1.eq(val2),
43            (Literal::Block(val1), Literal::Block(val2)) => val1 == val2,
44            _ => false,
45        }
46    }
47}
48
49impl Eq for Literal {}
50
51impl Hash for Literal {
52    fn hash<H: Hasher>(&self, state: &mut H) {
53        match self {
54            Literal::Symbol(val) => {
55                state.write(b"sym#");
56                val.hash(state);
57            }
58            Literal::String(val) => {
59                state.write(b"string#");
60                val.hash(state);
61            }
62            Literal::Double(val) => {
63                state.write(b"dbl#");
64                val.to_bits().hash(state);
65            }
66            Literal::Integer(val) => {
67                state.write(b"int#");
68                val.hash(state);
69            }
70            Literal::BigInteger(val) => {
71                state.write(b"bigint#");
72                val.hash(state);
73            }
74            Literal::Array(val) => {
75                state.write(b"array#");
76                for elem in val.iter() {
77                    elem.hash(state)
78                }
79            }
80            Literal::Block(val) => {
81                state.write(b"blk");
82                val.hash(state);
83            }
84        }
85    }
86}

88pub fn value_from_literal(literal: &Literal, gc_interface: &mut GCInterface) -> Value {
89    match literal {
90        Literal::Symbol(sym) => Value::Symbol(*sym),
91        Literal::String(val) => Value::String(val.clone()),
92        Literal::Double(val) => Value::Double(*val),
93        Literal::Integer(val) => Value::Integer(*val),
94        Literal::BigInteger(val) => Value::BigInteger(val.clone()),
95        Literal::Array(val) => {
96            let arr = &val.iter().map(|lit| value_from_literal(lit, gc_interface)).collect::<Vec<_>>();
97            Value::Array(VecValue(gc_interface.alloc_slice(arr)))
98        }
99        Literal::Block(val) => Value::Block(val.clone()),
100    }
101}
