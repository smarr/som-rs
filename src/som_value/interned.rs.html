1use std::fmt::Display;
2
3/// An interned string.
4///
5/// This is fast to move, clone and compare.
6///
7/// NB: this was originally a u32, which I think is more sensible. It's now a u16 to have Send
8/// bytecodes store an Interned reference directly without increasing the enum size. This is fine
9/// on our benchmarks, but sounds like a bad choice for very large systems?
10#[derive(Debug, Clone, Copy, Eq, PartialEq, Hash)]
11pub struct Interned(pub u16);
12
13// hack. we pretty much never want this, and instead to print the associated string.
14impl Display for Interned {
15    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
16        f.write_str(&format!("{}", self.0))
17    }
18}
