27/// The central data structure for the interpreter.
28///
29/// It represents the complete state of the interpreter, like the known class definitions,
30/// the string interner and the stack frames.
31pub struct Universe {
32    /// The string interner for symbols.
33    pub interner: Interner,
34    /// The known global bindings.
35    pub globals: HashMap<Interned, Value>,
36    /// The path to search in for new classes.
37    pub classpath: Vec<PathBuf>,
38    /// The current frame for the operation
39    pub current_frame: Gc<Frame>,
40    /// The interpreter's core classes.
41    pub core: CoreClasses<Gc<Class>>,
42    /// The time record of the universe's creation.
43    pub start_time: Instant,
44    /// GC interface
45    pub gc_interface: &'static mut GCInterface,
46}

61    /// Initialize the universe from the given classpath, and given a heap size
62    pub fn with_classpath_and_heap_size(classpath: Vec<PathBuf>, heap_size: usize) -> Result<Self, Error> {
63        let mut interner = Interner::with_capacity(200);
64        let mut globals: HashMap<Interned, Value> = HashMap::new();
65
66        let gc_interface = GCInterface::init(heap_size, get_callbacks_for_gc());
67
68        let mut core: CoreClasses<Gc<Class>> = CoreClasses::from_load_cls_fn(|name: &str, super_cls: Option<&Gc<Class>>| {
69            Self::load_system_class(classpath.as_slice(), name, super_cls.cloned(), gc_interface, &mut interner).unwrap()
70        });
71
72        // TODO: these can be removed for the most part - in the AST at least, we set a lot of super class relationships when loading system classes directly.
73        core.object_class.class().set_class(&core.metaclass_class);
74        core.object_class.class().set_super_class(&core.class_class);
75
76        set_super_class(&mut core.class_class, &core.object_class, &core.metaclass_class);
77        set_super_class(&mut core.metaclass_class.clone(), &core.class_class, &core.metaclass_class);
78        // initializeSystemClass(nilClass, objectClass, "Nil");
79        set_super_class(&mut core.nil_class, &core.object_class, &core.metaclass_class);
80        // initializeSystemClass(arrayClass, objectClass, "Array");
81        set_super_class(&mut core.array_class, &core.object_class, &core.metaclass_class);
82        // initializeSystemClass(methodClass, arrayClass, "Method");
83        set_super_class(&mut core.method_class, &core.array_class, &core.metaclass_class);
84        // initializeSystemClass(stringClass, objectClass, "String");
85        set_super_class(&mut core.string_class, &core.object_class, &core.metaclass_class);
86        // initializeSystemClass(symbolClass, stringClass, "Symbol");
87        set_super_class(&mut core.symbol_class, &core.string_class, &core.metaclass_class);
88        // initializeSystemClass(integerClass, objectClass, "Integer");
89        set_super_class(&mut core.integer_class, &core.object_class, &core.metaclass_class);
90        // initializeSystemClass(primitiveClass, objectClass, "Primitive");
91        set_super_class(&mut core.primitive_class, &core.object_class, &core.metaclass_class);
92        // initializeSystemClass(doubleClass, objectClass, "Double");
93        set_super_class(&mut core.double_class, &core.object_class, &core.metaclass_class);
94
95        set_super_class(&mut core.system_class, &core.object_class, &core.metaclass_class);
96
97        set_super_class(&mut core.block_class, &core.object_class, &core.metaclass_class);
98        set_super_class(&mut core.block1_class, &core.block_class, &core.metaclass_class);
99        set_super_class(&mut core.block2_class, &core.block_class, &core.metaclass_class);
100        set_super_class(&mut core.block3_class, &core.block_class, &core.metaclass_class);
101
102        set_super_class(&mut core.boolean_class, &core.object_class, &core.metaclass_class);
103        set_super_class(&mut core.true_class, &core.boolean_class, &core.metaclass_class);
104        set_super_class(&mut core.false_class, &core.boolean_class, &core.metaclass_class);
105
106        for (cls_name, core_cls) in core.iter() {
107            globals.insert(interner.intern(cls_name), Value::Class(core_cls.clone()));
108        }
109
110        globals.insert(interner.intern("true"), Value::Boolean(true));
111        globals.insert(interner.intern("false"), Value::Boolean(false));
112        globals.insert(interner.intern("nil"), Value::NIL);
113
114        let system_instance = Value::Instance(gc_interface.alloc(Instance::from_class(core.system_class())));
115        globals.insert(interner.intern("system"), system_instance);
116
117        Ok(Self {
118            globals,
119            interner,
120            classpath,
121            current_frame: Gc::default(),
122            start_time: Instant::now(),
123            core,
124            gc_interface,
125        })
126    }

128    /// Load a class from its name into this universe.
129    pub fn load_class(&mut self, class_name: impl Into<String>) -> Result<Gc<Class>, Error> {
130        let class_name = class_name.into();
131
132        for path in &self.classpath {
133            let mut path = path.join(class_name.as_str());
134            path.set_extension("som");
135
136            // Read file contents.
137            let contents = match fs::read_to_string(path.as_path()) {
138                Ok(contents) => contents,
139                Err(_) => continue,
140            };
141
142            // Collect all tokens from the file.
143            let tokens: Vec<_> = som_lexer::Lexer::new(contents.as_str()).skip_comments(true).skip_whitespace(true).collect();
144
145            // Parse class definition from the tokens.
146            let defn = match som_parser::parse_file(tokens.as_slice()) {
147                Some(defn) => defn,
148                None => continue,
149            };
150
151            if defn.name != class_name {
152                return Err(anyhow!("{}: class name is different from file name.", path.display(),));
153            }
154
155            let super_class = if let Some(ref super_class) = defn.super_class {
156                let symbol = self.intern_symbol(super_class.as_str());
157                self.lookup_global(symbol).and_then(Value::as_class).unwrap_or_else(|| self.load_class(super_class).unwrap())
158            } else {
159                self.core.object_class.clone()
160            };
161
162            let mut class = Class::from_class_def(defn, Some(super_class.clone()), self.gc_interface, &mut self.interner).map_err(Error::msg)?;
163            set_super_class(&mut class, &super_class, &self.core.metaclass_class);
164
165            let symbol = self.intern_symbol(class.name());
166            self.globals.insert(symbol, Value::Class(class.clone()));
167
168            return Ok(class);
169        }
170
171        Err(anyhow!("could not find the '{}' class", class_name))
172    }

174    /// Load a system class (with an incomplete hierarchy).
175    pub fn load_system_class(
176        classpath: &[impl AsRef<Path>],
177        class_name: impl Into<String>,
178        super_class: Option<Gc<Class>>,
179        gc_interface: &mut GCInterface,
180        interner: &mut Interner,
181    ) -> Result<Gc<Class>, Error> {
182        let class_name = class_name.into();
183        for path in classpath {
184            let mut path = path.as_ref().join(class_name.as_str());
185            path.set_extension("som");
186
187            // Read file contents.
188            let contents = match fs::read_to_string(path.as_path()) {
189                Ok(contents) => contents,
190                Err(err) if err.kind() == io::ErrorKind::NotFound => continue,
191                Err(err) => return Err(Error::from(err)),
192            };
193
194            // Collect all tokens from the file.
195            let tokens: Vec<_> = som_lexer::Lexer::new(contents.as_str()).skip_comments(true).skip_whitespace(true).collect();
196
197            // Parse class definition from the tokens.
198            let defn = match som_parser::parse_file_no_universe(tokens.as_slice()) {
199                Some(defn) => defn,
200                None => return Err(anyhow!("could not parse the '{}' system class", class_name)),
201            };
202
203            if defn.name != class_name {
204                return Err(anyhow!("{}: class name is different from file name.", path.display(),));
205            }
206
207            return Class::from_class_def(defn, super_class, gc_interface, interner).map_err(Error::msg);
208        }
209
210        Err(anyhow!("could not find the '{}' system class", class_name))
211    }

215    /// Evaluates a method or other after pushing a new frame onto the stack.
216    /// The frame assumes the arguments it needs are on the global argument stack.
217    pub fn eval_with_frame<T: Evaluate>(&mut self, value_stack: &mut GlobalValueStack, nbr_locals: u8, nbr_args: usize, invokable: &mut T) -> Return {
218        let frame = Frame::alloc_new_frame(nbr_locals, nbr_args, self, value_stack);
219        self.current_frame = frame;
220        let ret = invokable.evaluate(self, value_stack);
221        self.current_frame = self.current_frame.prev_frame.clone();
222        ret
223    }
224
225    /// Evaluates a block after pushing a new block frame.
226    pub fn eval_block_with_frame(&mut self, value_stack: &mut GlobalValueStack, nbr_locals: u8, nbr_args: usize) -> Return {
227        let frame = Frame::alloc_new_frame(nbr_locals, nbr_args, self, value_stack);
228        self.current_frame = frame.clone();
229        debug_assert_valid_semispace_ptr!(self.current_frame);
230        let mut invokable = frame.lookup_argument(0).as_block().unwrap();
231        debug_assert_valid_semispace_ptr!(invokable);
232        debug_assert_valid_semispace_ptr!(invokable.block);
233        let ret = invokable.evaluate(self, value_stack);
234        self.current_frame = self.current_frame.prev_frame.clone();
235        ret
236    }
237
238    /// Evaluates a block after pushing a new block frame, and the new frame doesn't get created by popping arguments - just copying them.
239    /// Implemented to help avoid an odd GC bug in the `to:by:do` node where a ref to a pointer, kept as a variable, ended up dangling after GC for some reason.
240    /// This variable was popped off the stack and pushed back. Thanks to this function, it's just copied from the previous one to the new one. Which may also be a speedup
241    pub fn eval_block_with_frame_no_pop(&mut self, value_stack: &mut GlobalValueStack, nbr_locals: u8, nbr_args: usize) -> Return {
242        let frame = Frame::alloc_new_frame_no_pop(nbr_locals, nbr_args, self, value_stack);
243        self.current_frame = frame.clone();
244        debug_assert_valid_semispace_ptr!(self.current_frame);
245        let mut invokable = frame.lookup_argument(0).as_block().unwrap();
246        debug_assert_valid_semispace_ptr!(invokable);
247        debug_assert_valid_semispace_ptr!(invokable.block);
248        let ret = invokable.evaluate(self, value_stack);
249        self.current_frame = self.current_frame.prev_frame.clone();
250        ret
251    }

289impl GlobalValueStack {
290    /// Standard push-to-stack operation. Exists so we can check pointers in debug mode, really
291    pub fn push(&mut self, value: Value) {
292        debug_assert_valid_semispace_ptr_value!(value);
293        self.0.push(value);
294    }
295
296    /// Standard pop.
297    pub fn pop(&mut self) -> Value {
298        debug_assert!(!self.0.is_empty());
299        // unsafe in the holy name of performance (preach) (I hope this is an OK speedup)
300        unsafe { self.0.pop().unwrap_unchecked() }
301    }
302
303    pub fn last(&mut self) -> &Value {
304        self.0.last().unwrap()
305    }
306
307    pub fn is_empty(&self) -> bool {
308        self.0.is_empty()
309    }
310
311    /// Remove N elements off the argument stack and return them as their own vector.
312    pub fn split_off_n(&mut self, n: usize) -> Vec<Value> {
313        let idx_split_off = self.0.len() - n;
314        self.0.split_off(idx_split_off)
315    }
316
317    /// Return the last `n` elements as a `Drain` iterator.
318    /// Faster than splitting off, at least for our code.
319    pub fn drain_n_last(&mut self, n: usize) -> Drain<'_, Value> {
320        let idx_split_off = self.0.len() - n;
321        self.0.drain(idx_split_off..)
322    }
323
324    pub fn borrow_n_last(&self, n: usize) -> &[Value] {
325        let idx_split_off = self.0.len() - n;
326        &self.0.as_slice()[idx_split_off..]
327    }
328
329    pub fn remove_n_last(&mut self, n: usize) {
330        let new_len = self.0.len() - n;
331        self.0.truncate(new_len)
332    }
333
334    pub fn iter(&self) -> Iter<'_, Value> {
335        self.0.iter()
336    }
337}

339impl Universe {
340    /// Call `escapedBlock:` on the given value, if it is defined.
341    pub fn escaped_block(&mut self, value_stack: &mut GlobalValueStack, value: Value, block: Gc<Block>) -> Option<Return> {
342        let method_name = self.intern_symbol("escapedBlock:");
343        let mut initialize = value.lookup_method(self, method_name)?;
344
345        value_stack.push(value);
346        value_stack.push(Value::Block(block));
347        let escaped_block_result = initialize.invoke(self, value_stack, 2);
348        Some(escaped_block_result)
349    }
350
351    /// Call `doesNotUnderstand:` on the given value, if it is defined.
352    pub fn does_not_understand(
353        &mut self,
354        value_stack: &mut GlobalValueStack,
355        value: Value,
356        interned_sym: Interned,
357        args: Vec<Value>,
358    ) -> Option<Return> {
359        let method_name = self.intern_symbol("doesNotUnderstand:arguments:");
360        let mut initialize = value.lookup_method(self, method_name)?;
361        let sym = Value::Symbol(interned_sym);
362        let args = Value::Array(VecValue(self.gc_interface.alloc_slice(&args)));
363
364        //eprintln!("Couldn't invoke {}; exiting.", self.interner.lookup(interned_sym));
365        //std::process::exit(1);
366
367        value_stack.push(value);
368        value_stack.push(sym);
369        value_stack.push(args);
370
371        let dnu_result = initialize.invoke(self, value_stack, 3);
372        Some(dnu_result)
373    }
374
375    /// Call `unknownGlobal:` on the given value, if it is defined.
376    pub fn unknown_global(&mut self, value_stack: &mut GlobalValueStack, value: Value, sym: Interned) -> Option<Return> {
377        let method_name = self.intern_symbol("unknownGlobal:");
378        let mut method = value.lookup_method(self, method_name)?;
379
380        value_stack.push(value);
381        value_stack.push(Value::Symbol(sym));
382
383        let unknown_global_result = method.invoke(self, value_stack, 2);
384        match unknown_global_result {
385            Return::Local(value) | Return::NonLocal(value, _) => Some(Return::Local(value)),
386            #[cfg(feature = "inlining-disabled")]
387            Return::Restart => panic!("(from 'System>>#unknownGlobal:') incorrectly asked for a restart"),
388        }
389    }
390
391    /// Call `System>>#initialize:` with the given name, if it is defined.
392    pub fn initialize(&mut self, args: Vec<Value>, value_stack: &mut GlobalValueStack) -> Option<Return> {
393        let method_name = self.interner.intern("initialize:");
394        let mut initialize = self.core.system_class().lookup_method(method_name)?;
395        let args = Value::Array(VecValue(self.gc_interface.alloc_slice(&args)));
396
397        let system_value = self.lookup_global(self.interner.reverse_lookup("system")?)?;
398        value_stack.push(system_value);
399
400        value_stack.push(args);
401        let program_result = initialize.invoke(self, value_stack, 2);
402        Some(program_result)
403    }
