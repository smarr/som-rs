23/// Represents a loaded class.
24#[derive(Clone)]
25pub struct Class {
26    /// The class' name.
27    pub name: String,
28    /// The class of this class.
29    pub class: Gc<Class>,
30    /// The superclass of this class.
31    pub super_class: Option<Gc<Class>>,
32    /// The class' fields.
33    pub fields: Vec<Value>,
34    /// The class' fields names.
35    pub field_names: Vec<String>,
36    /// The class' methods/invokables.
37    pub methods: IndexMap<Interned, Gc<Method>>,
38    /// Is this class a static one ?
39    pub is_static: bool,
40}
41
42// I don't test every field, but this should be good enough, AFAIK.
43impl PartialEq for Class {
44    fn eq(&self, other: &Self) -> bool {
45        self.name == other.name
46            && self.fields == other.fields
47            && self.field_names == other.field_names
48            && self.methods == other.methods
49            && self.is_static == other.is_static
50    }
51}

53impl Class {
54    /// Load up a class from its class definition from the AST.
55    /// NB: super_class is only ever None for one class: the core Object class, which all other classes inherit from.
56    /// NB: while it takes the super_class as argument, it's not in charge of hooking it up to the class itself. That's `set_super_class`. Might need changing for clarity.
57    pub fn from_class_def(
58        defn: ClassDef,
59        super_class: Option<Gc<Class>>,
60        gc_interface: &mut GCInterface,
61        interner: &mut Interner,
62    ) -> Result<Gc<Class>, String> {
63        let static_locals = {
64            let mut static_locals = IndexMap::new();
65            for field in defn.static_locals.iter() {
66                if static_locals.insert(field.clone(), Value::NIL).is_some() {
67                    return Err(format!("{}: the field named '{}' is already defined in this class", defn.name, field,));
68                }
69            }
70            static_locals
71        };
72
73        let instance_locals = {
74            let mut instance_locals = IndexMap::new();
75            for field in defn.instance_locals.iter() {
76                if instance_locals.insert(field.clone(), Value::NIL).is_some() {
77                    return Err(format!("{}: the field named '{}' is already defined in this class", defn.name, field,));
78                }
79            }
80            instance_locals
81        };
82
83        let maybe_static_superclass = super_class.clone().map(|cls| cls.class.clone());
84
85        let static_class = Self {
86            name: format!("{} class", defn.name),
87            class: Gc::default(),
88            super_class: maybe_static_superclass,
89            fields: vec![Value::NIL; static_locals.len()],
90            field_names: defn.static_locals,
91            methods: IndexMap::new(),
92            is_static: true,
93        };
94
95        let mut static_class_gc_ptr = gc_interface.alloc(static_class);
96
97        let instance_class = Self {
98            name: defn.name.clone(),
99            class: static_class_gc_ptr.clone(),
100            super_class,
101            fields: vec![Value::NIL; instance_locals.len()],
102            field_names: defn.instance_locals,
103            methods: IndexMap::new(),
104            is_static: false,
105        };
106
107        let mut instance_class_gc_ptr = gc_interface.alloc(instance_class);
108
109        let mut static_methods: IndexMap<Interned, Gc<Method>> = defn
110            .static_methods
111            .iter()
112            .map(|method| {
113                let signature = method.signature.clone();
114                let kind = AstMethodCompilerCtxt::get_method_kind(method, Some(static_class_gc_ptr.clone()), gc_interface, interner);
115                let method = Method {
116                    kind,
117                    signature: signature.clone(),
118                    holder: static_class_gc_ptr.clone(),
119                };
120                (interner.intern(signature.as_str()), gc_interface.alloc(method))
121            })
122            .collect();
123
124        if let Some(primitives) = primitives::get_class_primitives(&defn.name) {
125            for (signature, primitive, _warning) in primitives {
126                let interned_signature = interner.intern(signature);
127
128                //if *warning && !static_methods.contains_key(&interned_signature) {
129                //    eprintln!("Warning: Primitive '{}' is not in class definition for class '{}'", signature, defn.name);
130                //}
131
132                let method = Method {
133                    kind: MethodKind::Primitive(*primitive),
134                    signature: signature.to_string(),
135                    holder: static_class_gc_ptr.clone(),
136                };
137                static_methods.insert(interned_signature, gc_interface.alloc(method));
138            }
139        }
140
141        let mut instance_methods: IndexMap<Interned, Gc<Method>> = defn
142            .instance_methods
143            .iter()
144            .map(|method| {
145                let interned_signature = interner.intern(&method.signature);
146                let kind = AstMethodCompilerCtxt::get_method_kind(method, Some(instance_class_gc_ptr.clone()), gc_interface, interner);
147                let method = Method {
148                    kind,
149                    signature: method.signature.clone(),
150                    holder: instance_class_gc_ptr.clone(),
151                };
152                (interned_signature, gc_interface.alloc(method))
153            })
154            .collect();
155
156        if let Some(primitives) = primitives::get_instance_primitives(&defn.name) {
157            for (signature, primitive, _warning) in primitives {
158                let interned_signature = interner.intern(signature);
159
160                //if *warning && !instance_methods.contains_key(&interned_signature) {
161                //    eprintln!("Warning: Primitive '{}' is not in class definition for class '{}'", signature, defn.name);
162                //}
163
164                let method = Method {
165                    kind: MethodKind::Primitive(*primitive),
166                    signature: signature.to_string(),
167                    holder: instance_class_gc_ptr.clone(),
168                };
169                instance_methods.insert(interned_signature, gc_interface.alloc(method));
170            }
171        }
172
173        static_class_gc_ptr.methods = static_methods;
174        instance_class_gc_ptr.methods = instance_methods; // todo does this work? remove if runs ok
175
176        Ok(instance_class_gc_ptr)
177    }

179    /// Get the class' name.
180    pub fn name(&self) -> &str {
181        self.name.as_str()
182    }
183
184    /// Get the class of this class.
185    pub fn class(&self) -> Gc<Self> {
186        self.class.clone()
187    }
188
189    /// Set the class of this class (as a weak reference).
190    pub fn set_class(&mut self, class: &Gc<Self>) {
191        self.class = class.clone()
192    }
193
194    /// Get the superclass of this class.
195    pub fn super_class(&self) -> Option<Gc<Self>> {
196        self.super_class.clone()
197    }
198
199    /// Set the superclass of this class (as a weak reference).
200    pub fn set_super_class(&mut self, class: &Gc<Self>) {
201        // for local_name in class.borrow().field_names.iter().rev() {
202        //     self.field_names.insert(0, local_name.clone());
203        // }
204        for local in class.fields.iter().rev() {
205            self.fields.insert(0, *local);
206        }
207
208        self.super_class = Some(class.clone());
209    }
210
211    /// Search for a given method within this class.
212    pub fn lookup_method(&self, signature: Interned) -> Option<Gc<Method>> {
213        self.methods.get(&signature).cloned().or_else(|| self.super_class()?.lookup_method(signature))
214    }
215
216    /// Search for a local binding.
217    pub fn lookup_field(&self, idx: u8) -> Value {
218        self.fields.get(idx as usize).cloned().unwrap_or_else(|| {
219            let super_class = self.super_class().unwrap();
220            super_class.lookup_field(idx)
221        })
222    }
223
224    /// Assign a value to a local binding.
225    pub fn assign_field(&mut self, idx: u8, value: Value) {
226        if let Some(local) = self.fields.get_mut(idx as usize) {
227            *local = value;
228            return;
229        }
230        let mut super_class = self.super_class().unwrap();
231        super_class.assign_field(idx, value);
232    }
233
234    /// Used during parsing, to generate a FieldRead or a FieldWrite.
235    /// Iterates through superclasses to find the index of the field in a given class when it's originally defined in a superclass.
236    pub fn get_field_offset_by_name(&self, name: &str) -> Option<usize> {
237        self.field_names
238            .iter()
239            .position(|field_name| field_name == name)
240            .map(|pos| pos + self.super_class().map(|scls| scls.get_total_field_nbr()).unwrap_or(0))
241            .or_else(|| match self.super_class() {
242                Some(super_class) => super_class.get_field_offset_by_name(name),
243                _ => None,
244            })
245    }
246
247    pub fn get_total_field_nbr(&self) -> usize {
248        let scls_nbr_fields = match self.super_class() {
249            Some(scls) => scls.get_total_field_nbr(),
250            None => 0,
251        };
252        self.field_names.len() + scls_nbr_fields
253    }
254
255    /// Used by the `fields` primitive. Could be made faster (strings get cloned, then put on the GC heap in the primitive), but it's also basically never used.
256    pub fn get_all_field_names(&self) -> Vec<String> {
257        self.field_names
258            .iter()
259            .cloned()
260            .chain(self.super_class.as_ref().map(|scls| scls.get_all_field_names()).unwrap_or_default())
261            .collect()
262    }
263}
