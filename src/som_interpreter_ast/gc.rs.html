111fn get_roots_in_mutator_thread(_mutator: &mut Mutator<SOMVM>) -> Vec<SOMSlot> {
112    debug!("calling scan_roots_in_mutator_thread");
113    unsafe {
114        let mut to_process: Vec<SOMSlot> = vec![];
115
116        assert!(
117            !(*UNIVERSE_RAW_PTR_CONST.as_ptr()).is_null(),
118            "GC triggered while the system wasn't finished initializing."
119        );
120
121        // walk the frame list.
122        let current_frame_addr = &(**UNIVERSE_RAW_PTR_CONST.as_ptr()).current_frame;
123        debug!("scanning root: current_frame");
124        to_process.push(SOMSlot::from(current_frame_addr));
125
126        // walk globals (includes core classes, but we also need to move the refs in the CoreClasses class)
127        debug!("scanning roots: globals");
128        for (_name, val) in (**UNIVERSE_RAW_PTR_CONST.as_ptr()).globals.iter() {
129            visit_value_maybe_process(val, &mut to_process)
130        }
131
132        debug!("scanning roots: core classes");
133        for (_, cls_ptr) in (**UNIVERSE_RAW_PTR_CONST.as_ptr()).core.iter() {
134            to_process.push(SOMSlot::from(cls_ptr))
135        }
136
137        debug!("scanning roots: global argument stack");
138        for val in (**STACK_ARGS_RAW_PTR_CONST.as_ptr()).iter() {
139            visit_value_maybe_process(val, &mut to_process)
140        }
141
142        debug!("scanning roots: finished");
143        to_process
144    }
145}

147pub fn scan_object<'a>(object: ObjectReference, slot_visitor: &'a mut (dyn SlotVisitor<SOMSlot> + 'a)) {
148    unsafe {
149        let gc_id: &AstObjMagicId = VMObjectModel::ref_to_header(object).as_ref();
150
151        debug!("entering scan_object (type: {:?})", gc_id);
152
153        match gc_id {
154            AstObjMagicId::Frame => {
155                let frame: &Frame = object.to_raw_address().as_ref();
156
157                if !frame.prev_frame.is_empty() {
158                    slot_visitor.visit_slot(SOMSlot::from(&frame.prev_frame));
159                }
160
161                // kinda ew
162                let gcref_frame: Gc<Frame> = Gc::from(object.to_raw_address());
163
164                for i in 0..frame.nbr_locals {
165                    let val: &Value = gcref_frame.lookup_local(i);
166                    visit_value(val, slot_visitor)
167                }
168
169                for i in 0..frame.nbr_args {
170                    let val: &Value = gcref_frame.lookup_argument(i);
171                    visit_value(val, slot_visitor)
172                }
173            }
174            AstObjMagicId::Class => {
175                let class: &Class = object.to_raw_address().as_ref();
176
177                slot_visitor.visit_slot(SOMSlot::from(&class.class));
178
179                if let Some(scls) = class.super_class.as_ref() {
180                    slot_visitor.visit_slot(SOMSlot::from(scls));
181                }
182
183                for (_, method_ref) in class.methods.iter() {
184                    slot_visitor.visit_slot(SOMSlot::from(method_ref))
185                }
186
187                for field_ref in class.fields.iter() {
188                    visit_value(field_ref, slot_visitor)
189                }
190            }
191            AstObjMagicId::Method => {
192                let method: &Method = object.to_raw_address().as_ref();
193
194                slot_visitor.visit_slot(SOMSlot::from(&method.holder));
195
196                match &method.kind {
197                    MethodKind::Defined(method_def) => {
198                        for expr in &method_def.body.exprs {
199                            visit_expr(expr, slot_visitor)
200                        }
201                    }
202                    MethodKind::TrivialLiteral(trivial_lit) => visit_literal(&trivial_lit.literal, slot_visitor),
203                    MethodKind::TrivialGlobal(trivial_global) => {
204                        if let Some(cached_entry) = trivial_global.global_name.cached_entry.as_ref() {
205                            visit_value(cached_entry, slot_visitor)
206                        }
207                    }
208                    MethodKind::Primitive(_) | MethodKind::TrivialGetter(_) | MethodKind::TrivialSetter(_) => {}
209                }
210            }
211            AstObjMagicId::Instance => {
212                let instance: &Instance = object.to_raw_address().as_ref();
213
214                slot_visitor.visit_slot(SOMSlot::from(&instance.class));
215
216                for val in &instance.fields {
217                    visit_value(val, slot_visitor)
218                }
219            }
220            AstObjMagicId::Block => {
221                let block: &Block = object.to_raw_address().as_ref();
222                slot_visitor.visit_slot(SOMSlot::from(&block.frame));
223                slot_visitor.visit_slot(SOMSlot::from(&block.block));
224            }
225            AstObjMagicId::AstBlock => {
226                let ast_block: &AstBlock = object.to_raw_address().as_ref();
227                for expr in &ast_block.body.exprs {
228                    visit_expr(expr, slot_visitor)
229                }
230            }
231            AstObjMagicId::VecAstLiteral => {
232                let literal_vec: GcSlice<AstLiteral> = GcSlice::from(object.to_raw_address());
233                for lit in literal_vec.iter() {
234                    visit_literal(lit, slot_visitor)
235                }
236            }
237            AstObjMagicId::ArrayVal => {
238                let array_val: GcSlice<Value> = GcSlice::from(object.to_raw_address());
239                for val in array_val.iter() {
240                    visit_value(val, slot_visitor)
241                }
242            }
243            AstObjMagicId::String | AstObjMagicId::BigInt => {} // leaf nodes
244        }
245    }
246}

248/// Visits a value, via a specialized `SOMSlot` for value types.
249/// # Safety
250/// Values passed to this function MUST live on the GC heap, or the pointer generated from the reference will be invalid.
251unsafe fn visit_value<'a>(val: &Value, slot_visitor: &'a mut (dyn SlotVisitor<SOMSlot> + 'a)) {
252    if val.is_ptr_type() {
253        if let Some(slice) = val.as_array() {
254            // large object storage means no copying needed, but we still check the values stored
255            if slice.get_true_size() >= 65535 {
256                for val in slice.iter() {
257                    visit_value(val, slot_visitor)
258                }
259                return;
260            }
261        }
262        slot_visitor.visit_slot(SOMSlot::from(val.as_mut_ptr()))
263    }
264}

266/// Visits a value and potentially adds a slot made out of it to an array.
267/// # Safety
268/// Same as `visit_value`.
269unsafe fn visit_value_maybe_process(val: &Value, to_process: &mut Vec<SOMSlot>) {
270    if val.is_ptr_type() {
271        if let Some(slice) = val.as_array() {
272            // large object storage means no copying needed, but we still check the values stored
273            if slice.get_true_size() >= 65535 {
274                for val2 in slice.iter() {
275                    visit_value_maybe_process(val2, to_process);
276                }
277                return;
278            }
279        }
280        to_process.push(SOMSlot::from(val.as_mut_ptr()))
281    }
282}

284fn visit_expr(expr: &AstExpression, slot_visitor: &mut dyn SlotVisitor<SOMSlot>) {
285    fn visit_dispatch_node(dispatch_node: &AstDispatchNode, slot_visitor: &mut dyn SlotVisitor<SOMSlot>) {
286        visit_expr(&dispatch_node.receiver, slot_visitor);
287        if let Some(cache) = &dispatch_node.inline_cache {
288            slot_visitor.visit_slot(SOMSlot::from(&cache.0));
289            slot_visitor.visit_slot(SOMSlot::from(&cache.1));
290        }
291    }
292
293    match expr {
294        AstExpression::Block(blk) => slot_visitor.visit_slot(SOMSlot::from(blk)),
295        AstExpression::Literal(lit) => visit_literal(lit, slot_visitor),
296        AstExpression::InlinedCall(inlined_node) => match inlined_node.as_ref() {
297            InlinedNode::IfInlined(if_inlined) => {
298                visit_expr(&if_inlined.cond_expr, slot_visitor);
299                for expr in &if_inlined.body_instrs.exprs {
300                    visit_expr(expr, slot_visitor)
301                }
302            }
303            InlinedNode::IfNilInlined(if_nil_inlined) => {
304                visit_expr(&if_nil_inlined.cond_expr, slot_visitor);
305                for expr in &if_nil_inlined.body_instrs.exprs {
306                    visit_expr(expr, slot_visitor)
307                }
308            }
309            InlinedNode::IfTrueIfFalseInlined(if_true_if_false_inlined) => {
310                visit_expr(&if_true_if_false_inlined.cond_expr, slot_visitor);
311                for expr in &if_true_if_false_inlined.body_1_instrs.exprs {
312                    visit_expr(expr, slot_visitor)
313                }
314                for expr in &if_true_if_false_inlined.body_2_instrs.exprs {
315                    visit_expr(expr, slot_visitor)
316                }
317            }
318            InlinedNode::IfNilIfNotNilInlined(if_nil_if_not_nil_inlined) => {
319                visit_expr(&if_nil_if_not_nil_inlined.cond_expr, slot_visitor);
320                for expr in &if_nil_if_not_nil_inlined.body_1_instrs.exprs {
321                    visit_expr(expr, slot_visitor)
322                }
323                for expr in &if_nil_if_not_nil_inlined.body_2_instrs.exprs {
324                    visit_expr(expr, slot_visitor)
325                }
326            }
327            InlinedNode::WhileInlined(while_inlined) => {
328                for expr in &while_inlined.cond_instrs.exprs {
329                    visit_expr(expr, slot_visitor)
330                }
331                for expr in &while_inlined.body_instrs.exprs {
332                    visit_expr(expr, slot_visitor)
333                }
334            }
335            InlinedNode::OrInlined(or_inlined) => {
336                visit_expr(&or_inlined.first, slot_visitor);
337                for expr in &or_inlined.second.exprs {
338                    visit_expr(expr, slot_visitor)
339                }
340            }
341            InlinedNode::AndInlined(and_inlined) => {
342                visit_expr(&and_inlined.first, slot_visitor);
343                for expr in &and_inlined.second.exprs {
344                    visit_expr(expr, slot_visitor)
345                }
346            }
347            InlinedNode::ToDoInlined(to_do_inlined) => {
348                visit_expr(&to_do_inlined.start, slot_visitor);
349                visit_expr(&to_do_inlined.end, slot_visitor);
350                for expr in &to_do_inlined.body.exprs {
351                    visit_expr(expr, slot_visitor);
352                }
353            }
354        },
355        AstExpression::LocalExit(expr)
356        | AstExpression::NonLocalExit(expr, _)
357        | AstExpression::LocalVarWrite(_, expr)
358        | AstExpression::ArgWrite(_, _, expr)
359        | AstExpression::FieldWrite(_, expr)
360        | AstExpression::NonLocalVarWrite(_, _, expr) => visit_expr(expr, slot_visitor),
361        AstExpression::UnaryDispatch(dispatch) => {
362            visit_dispatch_node(&dispatch.dispatch_node, slot_visitor);
363        }
364        AstExpression::BinaryDispatch(dispatch) => {
365            visit_dispatch_node(&dispatch.dispatch_node, slot_visitor);
366            visit_expr(&dispatch.arg, slot_visitor)
367        }
368        AstExpression::TernaryDispatch(dispatch) => {
369            visit_dispatch_node(&dispatch.dispatch_node, slot_visitor);
370            visit_expr(&dispatch.arg1, slot_visitor);
371            visit_expr(&dispatch.arg2, slot_visitor);
372        }
373        AstExpression::NAryDispatch(dispatch) => {
374            visit_dispatch_node(&dispatch.dispatch_node, slot_visitor);
375            for arg in &dispatch.values {
376                visit_expr(arg, slot_visitor);
377            }
378        }
379        AstExpression::SuperMessage(super_message) => {
380            slot_visitor.visit_slot(SOMSlot::from(&super_message.super_class));
381            for arg in &super_message.values {
382                visit_expr(arg, slot_visitor);
383            }
384        }
385        AstExpression::GlobalRead(global_node) => {
386            if let Some(cached_entry) = global_node.cached_entry.as_ref() {
387                unsafe { visit_value(cached_entry, slot_visitor) }
388            }
389        }
390        AstExpression::LocalVarRead(..)
391        | AstExpression::NonLocalVarRead(..)
392        | AstExpression::IncLocal(..)
393        | AstExpression::DecLocal(..)
394        | AstExpression::ArgRead(..)
395        | AstExpression::FieldRead(..) => {} // leaf nodes
396    }
397}

399/// Visits a value, via a specialized `SOMSlot` for value types.
400/// # Safety
401/// Literals passed to this function MUST live on the GC heap, but that's always the case for literals (at the moment).
402fn visit_literal(literal: &AstLiteral, slot_visitor: &mut dyn SlotVisitor<SOMSlot>) {
403    match &literal {
404        AstLiteral::String(s) => slot_visitor.visit_slot(SOMSlot::from(s)),
405        AstLiteral::BigInteger(big_int) => slot_visitor.visit_slot(SOMSlot::from(big_int)),
406        AstLiteral::Array(arr) => slot_visitor.visit_slot(SOMSlot::from(arr)),
407        AstLiteral::Symbol(_) | AstLiteral::Double(_) | AstLiteral::Integer(_) => {}
408    }
409}

413fn get_object_size(object: ObjectReference) -> usize {
414    let gc_id: &AstObjMagicId = unsafe { VMObjectModel::ref_to_header(object).as_ref() };
415
416    match gc_id {
417        AstObjMagicId::Frame => unsafe {
418            let frame: &Frame = object.to_raw_address().as_ref();
419            Frame::get_true_size(frame.nbr_args, frame.nbr_locals)
420        },
421        AstObjMagicId::Instance => size_of::<Instance>(),
422        AstObjMagicId::String => size_of::<String>(),
423        AstObjMagicId::BigInt => size_of::<BigInt>(),
424        AstObjMagicId::AstBlock => size_of::<AstBlock>(),
425        AstObjMagicId::VecAstLiteral => {
426            let literals: GcSlice<AstLiteral> = GcSlice::from(object.to_raw_address());
427            literals.get_true_size()
428        }
429        AstObjMagicId::ArrayVal => {
430            let values: GcSlice<Value> = GcSlice::from(object.to_raw_address());
431            values.get_true_size()
432        }
433        AstObjMagicId::Method => size_of::<Method>(),
434        AstObjMagicId::Block => size_of::<Block>(),
435        AstObjMagicId::Class => size_of::<Class>(),
436    }
437}
