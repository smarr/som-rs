1use crate::invokable::{Invoke, Return};
2use crate::universe::{GlobalValueStack, Universe};
3use crate::value::Value;
4
5#[derive(Debug, Clone, PartialEq)]
6pub struct ToByDoNode {}
7
8impl Invoke for ToByDoNode {
9    fn invoke(&mut self, universe: &mut Universe, value_stack: &mut GlobalValueStack, nbr_args: usize) -> Return {
10        // let args = value_stack.pop_n_last(nbr_args);
11        let args = value_stack.borrow_n_last(nbr_args);
12        let start_int_val = &args[0];
13        let step_int_val = &args[1];
14        let end_int_val = &args[2];
15        let body_block_val = &args[3];
16
17        let (start_int, end_int, step_int, body_block) = match (
18            start_int_val.as_integer(),
19            step_int_val.as_integer(),
20            end_int_val.as_integer(),
21            body_block_val.as_block(),
22        ) {
23            (Some(a), Some(b), Some(c), Some(d)) => (a, b, c, d),
24            (a, b, c, d) => panic!(
25                "to:by:do: was not given three ints and a block as arguments, but {:?} and {:?} and {:?} and {:?}",
26                a, b, c, d
27            ),
28        };
29
30        let nbr_locals = body_block.block.nbr_locals;
31        let mut i = start_int;
32        while i <= end_int {
33            value_stack.push(Value::Integer(i));
34
35            match universe.eval_block_with_frame_no_pop(value_stack, nbr_locals, 2) {
36                Return::Local(..) => {}
37                ret => {
38                    value_stack.remove_n_last(nbr_args + 1); // all the arguments, and also including the integer index
39                    return ret;
40                }
41            };
42
43            value_stack.pop();
44            i += step_int;
45        }
46
47        value_stack.remove_n_last(nbr_args);
48        Return::Local(Value::Integer(start_int))
49    }
50}
