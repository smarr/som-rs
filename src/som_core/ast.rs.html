1/// Represents a class definition.
2///
3/// Example:
4/// ```text
5/// Counter = (
6///   | total |
7///   new = ( self reset )
8///   increment = ( total := total + 1 )
9///   get = ( ^ total )
10///   reset = ( total := 0 )
11/// )
12/// ```
13#[derive(Debug, Clone, PartialEq)]
14pub struct ClassDef {
15    /// The name of the class.
16    pub name: String,
17    /// The name of the superclass.
18    pub super_class: Option<String>,
19    /// The locals for instances of that class.
20    pub instance_locals: Vec<String>,
21    /// The methods declared for instances of that class.
22    pub instance_methods: Vec<MethodDef>,
23    /// The static locals for that class.
24    pub static_locals: Vec<String>,
25    /// The static methods declared for that class.
26    pub static_methods: Vec<MethodDef>,
27}

29/// Represents a method definition.
30///
31/// Example:
32/// ```text
33/// "unary method"       increment = ( self increment: 1 )
34/// "positional method"  increment: value = ( total := total + value )
35/// "operator method"    + value = ( self increment: value )
36/// ```
37#[derive(Debug, Clone, PartialEq)]
38pub struct MethodDef {
39    /// The method's signature (eg. `println`, `at:put:` or `==`).
40    pub signature: String,
41    /// The method's body.
42    pub body: MethodBody,
43}
44
45/// Represents a method's body.
46///
47/// Exemple:
48/// ```text
49/// "primitive method body"
50/// printString: string = primitive
51///
52/// "actual method body, with a local"
53/// double: value = ( |clone| clone := double. ^ (double + clone) )
54/// ```
55#[derive(Debug, Clone, PartialEq)]
56pub enum MethodBody {
57    /// A primitive (meant to be implemented by the VM itself).
58    Primitive,
59    /// An actual body for the method, with locals.
60    Body {
61        locals_nbr: usize,
62        body: Body,
63        #[cfg(feature = "block-debug-info")]
64        debug_info: BlockDebugInfo,
65    },
66}

68/// Represents the contents of a body (within a term or block).
69///
70/// Exemple:
71/// ```text
72/// "body within a term"
73/// new = (
74///     local := counter + 5.
75///     (counter get) > 5
76/// )
77///
78/// "body within a block"
79/// [ :arg |
80///     local := counter + arg.
81///     arg * (counter get)
82/// ]
83/// ```
84#[derive(Debug, Clone, PartialEq)]
85pub struct Body {
86    /// The expressions in the body.
87    pub exprs: Vec<Expression>,
88    /// Is the last expression terminated with a period ?
89    pub full_stopped: bool,
90}

92/// Represents an expression.
93///
94/// Exemple:
95/// ```text
96/// "reference"          counter
97/// "assignment"         counter := 10
98/// "messsage send"      counter incrementBy: 5
99/// "binary operation"   counter <= 5
100/// "exit operation"     ^counter
101/// "literal"            'foo'
102/// "block"              [ :value | counter incrementBy: value ]
103/// ```
104#[derive(Debug, Clone, PartialEq)]
105pub enum Expression {
106    /// A reference to a binding (eg. `counter`).
107    GlobalRead(String),
108    /// This does NOT exist: this is a field write which will be resolved by the AST/BC compilers, or it's an error.
109    GlobalWrite(String, Box<Expression>),
110    /// Read of a local var.
111    LocalVarRead(usize),
112    /// Read of a nonlocal var.
113    NonLocalVarRead(usize, usize),
114    /// Read of an argument.
115    ArgRead(usize, usize),
116    /// An assignment to a binding (eg. `counter := 10`).
117    LocalVarWrite(usize, Box<Expression>),
118    NonLocalVarWrite(usize, usize, Box<Expression>),
119    ArgWrite(usize, usize, Box<Expression>),
120    /// A message send (eg. `counter incrementBy: 5`).
121    Message(Box<Message>),
122    /// An exit operation (eg. `^counter`). Second argument is the scope level to differentiate local and nonlocal returns
123    Exit(Box<Expression>, usize),
124    /// A literal (eg. `'foo'`, `10`, `#foo`, ...).
125    Literal(Literal),
126    /// A block (eg. `[ :value | counter incrementBy: value ]`).
127    Block(Block),
128}

130/// Represents a message send.
131///
132/// Exemple:
133/// ```text
134/// "unary message send"
135/// 'hello, world' println
136///
137/// "positional message send"
138/// range from: 0 to: 10
139///
140/// "binary operator message send"
141/// value == 3
142/// ```
143#[derive(Debug, Clone, PartialEq)]
144pub struct Message {
145    /// The object to which the message is sent to.
146    pub receiver: Expression,
147    /// The signature of the message (eg. "ifTrue:ifFalse:").
148    pub signature: String,
149    /// The list of dynamic values that are passed.
150    pub values: Vec<Expression>,
151}
152
153/// A message with "super" as the receiver, so the superclass.
154#[derive(Debug, Clone, PartialEq)]
155pub struct SuperMessage {
156    /// The object to which the message is sent to. TODO should not do a super call, but cache the class, really.
157    pub receiver_name: String,
158    /// Do we access the static or instance methods of the superclass?
159    pub is_static_class_call: bool,
160    /// The signature of the message (eg. "ifTrue:ifFalse:").
161    pub signature: String,
162    /// The list of dynamic values that are passed.
163    pub values: Vec<Expression>,
164}

166/// Represents a binary operation.
167///
168/// Exemple:
169/// ```text
170/// counter <= 2
171/// ```
172#[derive(Debug, Clone, PartialEq)]
173pub struct BinaryOp {

175    pub op: String,

182/// Represents a block.
183///
184/// Exemple:
185/// ```text
186/// "simple block"
187/// [ 'hello, world' println ]
188///
189/// "block with parameter"
190/// [ :value | value * 2 ]
191///
192/// "block with parameter and local"
193/// [ :value | |serialized| serialized := value asString. serialized println ]
194/// ```
195#[derive(Debug, Clone, PartialEq)]
196pub struct Block {
197    /// Represents the parameters' names.
198    pub nbr_params: usize,
199    /// The names of the locals.
200    pub nbr_locals: usize,
201    /// Represents the block's body.
202    pub body: Body,
203    #[cfg(feature = "block-debug-info")]
204    /// Debug info for the block: parameters and local variable names
205    pub dbg_info: BlockDebugInfo,
206}
207
208#[cfg(feature = "block-debug-info")]
209#[derive(Debug, Clone, PartialEq)]
210pub struct BlockDebugInfo {
211    pub parameters: Vec<String>,
212    pub locals: Vec<String>,
213}

228/// Represents a literal.
229///
230/// Exemple:
231/// ```text
232/// #foo     "symbol literal"
233/// 'hello'  "string literal"
234/// 3.14     "double literal"
235/// 42       "integer literal"
236/// ```
237#[derive(Debug, Clone, PartialEq)]
238pub enum Literal {
239    /// Represents a symbol literal (eg. `#foo`).
240    Symbol(String), // todo: in a perfect world, this would be 'static str. that requires we store source code as static though, which sounds like a lot of refactoring.
241    /// Represents a string literal (eg. `'hello'`).
242    String(String), // todo: ditto, maybe.
243    /// Represents a decimal number literal (eg. `3.14`).
244    Double(f64),
245    /// Represents a integer number literal (eg. `42`).
246    Integer(i32),
247    /// Represents a big integer (bigger than a 64-bit signed integer can represent).
248    BigInteger(String),
249    /// Represents an array literal (eg. `$(1 2 3)`)
250    Array(Vec<Literal>),
251}
